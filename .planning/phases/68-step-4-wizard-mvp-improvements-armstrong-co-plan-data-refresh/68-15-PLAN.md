       import { NextRequest, NextResponse } from 'next/server'

       import type { OnePagePlanData } from '@/app/one-page-plan/types'

       export async function POST(req: NextRequest) {
         try {
           const supabase = await createClient()
           const { data: { user } } = await supabase.auth.getUser()
           if (!user) return NextResponse.json({ ok: false, error: 'unauthenticated' }, { status: 401 })

           const body = await req.json().catch(() => null)
           if (!body || typeof body.business_id !== 'string' || !body.step4_plan_data) {
             return NextResponse.json({ ok: false, error: 'business_id and step4_plan_data required' }, { status: 400 })
           }
           const { business_id, step4_plan_data, label: labelOverride } = body as {
             business_id: string
             step4_plan_data: Partial<OnePagePlanData>
             label?: string
           }

           // Authorization gate (RLS-backed):
           const { data: bp, error: bpErr } = await supabase
             .from('business_profiles')
             .select('id, company_name, owner_info')   // also pull data we need for merge
             .eq('id', business_id)
             .maybeSingle()
           if (bpErr) return NextResponse.json({ ok: false, error: bpErr.message }, { status: 500 })
           if (!bp) return NextResponse.json({ ok: false, error: 'business not found' }, { status: 404 })

           // SERVER-SIDE COMPOSITION (mirrors scripts/68-08-armstrong-plan-snapshot-baseline.mjs):
           const [strategyRes, swotItemsRes, businessRes] = await Promise.all([
             supabase.from('strategy_data').select('vision_mission').eq('business_id', business_id).maybeSingle(),
             // SWOT read — follow the existing repo pattern: locate swot_analyses for this business,
             // then read swot_items by swot_analysis_id. The actual join column MUST be verified
             // during read_first (do NOT guess). Pseudocode:
             supabase
               .from('swot_items')
               .select('category, content, swot_analyses!inner(business_id)')
               .eq('swot_analyses.business_id', business_id),
             // businesses lookup — verify the linkage column from business_profiles to businesses;
             // typical patterns are business_profiles.business_id or businesses.profile_id. Use
             // whichever the repo actually uses (verify via grep in read_first).
             supabase.from('businesses').select('name').eq('id', /* linkage from bp */ business_id).maybeSingle(),
           ])

           const vm = (strategyRes.data as any)?.vision_mission || {}
           const swotItems = (swotItemsRes.data as any[]) || []
           const businessName = (businessRes.data as any)?.name || bp.company_name || ''
           const ownerInfo = (bp as any).owner_info || {}

           const ownerGoals = {
             desiredHoursPerWeek: ownerInfo.desired_hours,
             currentHoursPerWeek: ownerInfo.current_hours_per_week,
             primaryGoal:         ownerInfo.primary_goal,
             timeHorizon:         ownerInfo.time_horizon,
             exitStrategy:        ownerInfo.exit_strategy,
           }

           const groupSwot = (cat: string) =>
             swotItems.filter((i: any) => i.category === cat).map((i: any) => i.content)

           const merged: OnePagePlanData = {
             vision:     vm.vision_statement || '',
             mission:    vm.mission_statement || '',
             coreValues: Array.isArray(vm.core_values) ? vm.core_values : [],
             strengths:     groupSwot('strengths'),
             weaknesses:    groupSwot('weaknesses'),
             opportunities: groupSwot('opportunities'),
             threats:       groupSwot('threats'),
             companyName: businessName,
             ownerGoals,
             ...step4_plan_data,             // Step-4 fields win for anything we both compute
           } as OnePagePlanData

           // Compute next version_number
           const { data: maxRow, error: maxErr } = await supabase
             .from('plan_snapshots')
             .select('version_number')
             .eq('business_id', business_id)
             .order('version_number', { ascending: false })
             .limit(1)
             .maybeSingle()
           if (maxErr) return NextResponse.json({ ok: false, error: maxErr.message }, { status: 500 })
           const next_version = (maxRow?.version_number ?? 0) + 1

           const label = labelOverride || `Wizard snapshot ${new Date().toISOString().slice(0,16).replace('T',' ')}`

           const { data: inserted, error: insErr } = await supabase
             .from('plan_snapshots')
             .insert({
               business_id,
               user_id: user.id,
               snapshot_type: 'goals_wizard_complete',
               version_number: next_version,
               plan_data: merged,
               label,
             })
             .select('id, version_number, label')
             .single()
           if (insErr) return NextResponse.json({ ok: false, error: insErr.message }, { status: 500 })

           return NextResponse.json({ ok: true, id: inserted.id, version_number: inserted.version_number, label: inserted.label })
         } catch (e: any) {
           return NextResponse.json({ ok: false, error: e?.message || 'unknown' }, { status: 500 })
         }
       }
