POST /api/plan-snapshots
  Request body (JSON):
    {
      business_id: string,    // business_profiles.id (same as plan_snapshots uses)
      user_id: string,        // current authenticated user
      label?: string,         // optional override; default 'Wizard snapshot YYYY-MM-DD HH:MM'
      plan_data: OnePagePlanData  // composed by client
    }

  Response (200):
    { ok: true, id: string, version_number: number, label: string }

  Response (4xx/5xx):
    { ok: false, error: string }

Server logic:
  1. Auth check (require authenticated user; reject 401 otherwise).
  2. Resolve user → confirm user has access to business_id (via existing resolveBusinessId pattern OR direct ownership check using the same auth-aware Supabase client).
  3. Read max version_number for (business_id) — use SUPABASE_SERVICE_KEY or authenticated client per repo conventions (check existing /api routes that write plan_snapshots, if any; otherwise mirror pattern from /api/quarterly-review or similar).
  4. INSERT row: { business_id, user_id, snapshot_type: 'goals_wizard_complete', version_number: max+1, plan_data, label }.
  5. Return new row id + version_number.

Client side:
  composedPlanData() returns an OnePagePlanData from current Step4Props (financialData, coreMetrics, kpis, twelveMonthInitiatives, annualPlanByQuarter, quarterlyTargets, yearType, planYear, businessId). Where data is missing, defaults are: empty arrays, 0 numbers. quarterlyRocks: [].

       ```
       import { NextRequest, NextResponse } from 'next/server'
       import { createClient } from '@/lib/supabase/server' // confirm path via repo; if different in this codebase, use the actual server client factory
       import type { OnePagePlanData } from '@/app/one-page-plan/types'

       export async function POST(req: NextRequest) {
         try {
           const supabase = await createClient()
           const { data: { user } } = await supabase.auth.getUser()
           if (!user) return NextResponse.json({ ok: false, error: 'unauthenticated' }, { status: 401 })

           const body = await req.json().catch(() => null)
           if (!body || typeof body.business_id !== 'string' || !body.plan_data) {
             return NextResponse.json({ ok: false, error: 'business_id and plan_data required' }, { status: 400 })
           }
           const { business_id, plan_data, label: labelOverride } = body as { business_id: string; plan_data: OnePagePlanData; label?: string }

           // Authorization: confirm the user has access to this business_id. Use the
           // existing authoritative pattern. If unsure, query business_profiles for
           // either user_id=user.id matching OR via business_users membership.
           // (Mirror the pattern used in /api/cfo/report-status or similar.)
           const { data: bp, error: bpErr } = await supabase
             .from('business_profiles')
             .select('id')
             .eq('id', business_id)
             .maybeSingle()
           if (bpErr) return NextResponse.json({ ok: false, error: bpErr.message }, { status: 500 })
           if (!bp) return NextResponse.json({ ok: false, error: 'business not found' }, { status: 404 })
           // RLS will block unauthorized fetches; if we got bp, the user has read access.

           // Compute next version_number
           const { data: maxRow, error: maxErr } = await supabase
             .from('plan_snapshots')
             .select('version_number')
             .eq('business_id', business_id)
             .order('version_number', { ascending: false })
             .limit(1)
             .maybeSingle()
           if (maxErr) return NextResponse.json({ ok: false, error: maxErr.message }, { status: 500 })
           const next_version = (maxRow?.version_number ?? 0) + 1

           const label = labelOverride || `Wizard snapshot ${new Date().toISOString().slice(0,16).replace('T',' ')}`

           const { data: inserted, error: insErr } = await supabase
             .from('plan_snapshots')
             .insert({
               business_id,
               user_id: user.id,
               snapshot_type: 'goals_wizard_complete',
               version_number: next_version,
               plan_data,
               label,
             })
             .select('id, version_number, label')
             .single()
           if (insErr) return NextResponse.json({ ok: false, error: insErr.message }, { status: 500 })

           return NextResponse.json({ ok: true, id: inserted.id, version_number: inserted.version_number, label: inserted.label })
         } catch (e: any) {
           return NextResponse.json({ ok: false, error: e?.message || 'unknown' }, { status: 500 })
         }
       }
       ```
