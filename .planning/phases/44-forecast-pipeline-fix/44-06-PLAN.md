```sql
CREATE TABLE forecast_pl_lines (
  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  forecast_id uuid NOT NULL REFERENCES financial_forecasts(id) ON DELETE CASCADE,
  account_name text,
  account_code text,
  category text,                          -- 'revenue' | 'cogs' | 'opex' | 'team' | etc.

  is_manual boolean DEFAULT false,
  created_at timestamptz DEFAULT now(),
  updated_at timestamptz DEFAULT now()
);
```

NOTE: `forecast_pl_lines` keeps its WIDE format (one row per account, monthly_values in JSONB). Unlike `xero_pl_lines` (which moved to long format in 44-02), `forecast_pl_lines` stays wide because: (a) it's per-forecast not per-tenant — no concurrency issue requires natural-key uniqueness across months, (b) every wizard read site already expects the wide JSONB shape, (c) the volume is tiny (~50 rows × 36 months per forecast). This deliberate inconsistency is documented in the migration COMMENT.

<!-- Existing forecast_assumptions schema -->

```sql
CREATE TABLE forecast_assumptions (
  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  forecast_id uuid NOT NULL REFERENCES financial_forecasts(id) ON DELETE CASCADE UNIQUE,
  data jsonb DEFAULT '{}',                -- the entire assumptions tree the wizard mutates
  updated_at timestamptz DEFAULT now()
);
```

The wizard's current materialization logic (in `/api/forecast-wizard-v4/generate/route.ts:160-217`) does:
1. Compute derived `forecast_months` from assumptions (in JS).
2. UPDATE forecast_assumptions SET data, updated_at = now().
3. DELETE FROM forecast_pl_lines WHERE forecast_id = X AND is_manual = false.
4. INSERT INTO forecast_pl_lines (...) SELECT (...).

If step 3 or 4 fails, step 2 is already committed (Supabase JS .update()/.delete()/.insert() each open their own transaction). The fix per D-12 is to bundle 2-4 inside a single PL/pgSQL function body. The DERIVATION (step 1) stays in JS — passed in as `p_pl_lines jsonb` argument.

<!-- RPC convention from RESEARCH.md Pattern 1 -->

```sql
CREATE OR REPLACE FUNCTION save_assumptions_and_materialize(
  p_forecast_id uuid,
  p_assumptions jsonb,
  p_pl_lines jsonb       -- array of {account_name, account_code, category, forecast_months}
) RETURNS jsonb
LANGUAGE plpgsql
SECURITY DEFINER
AS $$
DECLARE
  v_now timestamptz := now();
  v_lines_count int;
BEGIN
  UPDATE forecast_assumptions SET data = p_assumptions, updated_at = v_now WHERE forecast_id = p_forecast_id;
  IF NOT FOUND THEN
    INSERT INTO forecast_assumptions (forecast_id, data, updated_at) VALUES (p_forecast_id, p_assumptions, v_now);
  END IF;

  DELETE FROM forecast_pl_lines WHERE forecast_id = p_forecast_id AND is_manual = false;

  INSERT INTO forecast_pl_lines (forecast_id, account_name, account_code, category, forecast_months, computed_at, is_manual)
  SELECT p_forecast_id, line->>'account_name', line->>'account_code', line->>'category',
         (line->'forecast_months')::jsonb, v_now, false
  FROM jsonb_array_elements(p_pl_lines) AS line;

  GET DIAGNOSTICS v_lines_count = ROW_COUNT;
  RETURN jsonb_build_object('forecast_id', p_forecast_id, 'computed_at', v_now, 'lines_count', v_lines_count);
END;
$$;
```

    Create `supabase/migrations/20260429000002_save_assumptions_and_materialize_rpc.sql`:

    ```sql
    BEGIN;

    -- Phase 44 D-12 — atomic save + materialize.
    --
    -- Replaces /api/forecast-wizard-v4/generate/route.ts:160-217 and /api/Xero/sync-forecast/route.ts
    -- materialization logic. PostgREST runs each RPC inside an implicit transaction; a failure in
    -- the INSERT step rolls back the UPDATE step too — atomicity guaranteed.
    --
    -- A single `now()` value is captured at function entry and used for BOTH
    -- forecast_assumptions.updated_at AND forecast_pl_lines.computed_at, so consumers'
    -- invariant `computed_at >= updated_at` cannot fire on millisecond clock skew (Pitfall 3).
    --
    -- Args:
    --   p_forecast_id  — financial_forecasts.id this save targets
    --   p_assumptions  — full assumptions JSONB tree (replaces forecast_assumptions.data)
    --   p_pl_lines     — JSONB array of {account_name, account_code, category, forecast_months}
    --                     to materialize as forecast_pl_lines (is_manual = false rows are replaced;
    --                     is_manual = true rows are preserved untouched)
    --
    -- Returns:
    --   jsonb { forecast_id, computed_at (ISO timestamp), lines_count (number inserted) }

    CREATE OR REPLACE FUNCTION save_assumptions_and_materialize(
      p_forecast_id uuid,
      p_assumptions jsonb,
      p_pl_lines jsonb
    ) RETURNS jsonb
    LANGUAGE plpgsql
    SECURITY DEFINER
    SET search_path = public
    AS $$
    DECLARE
      v_now timestamptz := now();
      v_lines_count int;
      v_assumptions_existed boolean;
    BEGIN
      -- 1. Upsert assumptions row.
      UPDATE forecast_assumptions
      SET data = p_assumptions, updated_at = v_now
      WHERE forecast_id = p_forecast_id;

      GET DIAGNOSTICS v_assumptions_existed = ROW_COUNT;

      IF NOT v_assumptions_existed THEN
        INSERT INTO forecast_assumptions (forecast_id, data, updated_at)
        VALUES (p_forecast_id, p_assumptions, v_now);
      END IF;

      -- 2. Replace materialised rows. is_manual = true rows (coach overrides) are preserved.
      DELETE FROM forecast_pl_lines
      WHERE forecast_id = p_forecast_id AND is_manual = false;

      -- 3. Insert new derived rows with computed_at = v_now.
      INSERT INTO forecast_pl_lines (
        forecast_id, account_name, account_code, category,
        forecast_months, computed_at, is_manual, created_at, updated_at
      )
      SELECT
        p_forecast_id,
        line->>'account_name',
        line->>'account_code',
        line->>'category',
        COALESCE((line->'forecast_months')::jsonb, '{}'::jsonb),
        v_now,
        false,
        v_now,
        v_now
      FROM jsonb_array_elements(p_pl_lines) AS line;

      GET DIAGNOSTICS v_lines_count = ROW_COUNT;

      RETURN jsonb_build_object(
        'forecast_id', p_forecast_id,
        'computed_at', v_now,
        'lines_count', v_lines_count
      );
    END;
    $$;

    -- Auth: SECURITY DEFINER runs with the function owner's privileges. We restrict EXECUTE to authenticated
    -- (so the wizard can call it via supabase.rpc) and revoke from PUBLIC.
    -- The function does NOT check forecast ownership — the calling API route MUST verify the user has
    -- access to p_forecast_id BEFORE calling. (Mirrors Phase 35's helper pattern.)
    REVOKE ALL ON FUNCTION save_assumptions_and_materialize(uuid, jsonb, jsonb) FROM PUBLIC;
    GRANT EXECUTE ON FUNCTION save_assumptions_and_materialize(uuid, jsonb, jsonb) TO authenticated;
    GRANT EXECUTE ON FUNCTION save_assumptions_and_materialize(uuid, jsonb, jsonb) TO service_role;

    COMMENT ON FUNCTION save_assumptions_and_materialize(uuid, jsonb, jsonb) IS
      'Phase 44 D-12 — atomic write of forecast_assumptions + forecast_pl_lines in one transaction. ' ||
      'Single now() applied to both tables. Authorize at API layer; this fn does not check forecast ownership.';

    COMMIT;
    ```
