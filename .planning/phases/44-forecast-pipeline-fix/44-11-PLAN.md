<!-- Existing cell-render contract in Step3RevenueCOGS.tsx -->

The current Step3 component has a `isActualMonth(monthKey)` check (per UI-SPEC pre-population: line ~33 of the file) that gates locked-actual vs editable-projected cell rendering. Phase 44 adds a parallel `isMissingMonth(monthKey, monthly_values)` check that takes precedence over both.

Render contract (per UI-SPEC):
```typescript
function renderCell(monthKey: string, line: { monthly_values: Record<string, number>; ... }) {
  if (isMissingMonth(monthKey, line.monthly_values)) {
    return <MissingCell monthKey={monthKey} />              // NEW — em-dash
  }
  if (isActualMonth(monthKey)) {
    return <LockedActualCell value={line.monthly_values[monthKey]} />  // existing
  }
  return <EditableProjectedCell ... />                       // existing
}

function isMissingMonth(monthKey: string, monthly_values: Record<string, number>): boolean {
  // CRITICAL: hasOwnProperty.call distinguishes "Xero returned 0" (real-zero) from "Xero didn't return this column" (missing)
  return !Object.prototype.hasOwnProperty.call(monthly_values, monthKey)
}
```

import { Tooltip } from '@/components/ui/Tooltip'

const MONTH_NAMES = ['Jan','Feb','Mar','Apr','May','Jun','Jul','Aug','Sep','Oct','Nov','Dec'] as const

function fmtMonYYYY(yyyymm: string): string {
  const [y, m] = yyyymm.split('-')
  return `${MONTH_NAMES[parseInt(m, 10) - 1]} ${y}`
}

function MissingCell({ monthKey }: { monthKey: string }) {

      >
        —
      </td>
    </Tooltip>
  )
}
