    ```typescript
    import { NextRequest, NextResponse } from 'next/server'
    import * as Sentry from '@sentry/nextjs'
    import { runSyncForAllBusinesses } from '@/lib/xero/sync-orchestrator'

    /**
     * Phase 44 D-11 — Nightly Xero P&L sync.
     *
     * Schedule: 0 16 * * * UTC.
     * - 02:00 AEDT during Australian summer (Apr 2026: AEST starts 1 Apr, so this runs 02:00 AEST in winter Apr-Sep).
     * - 03:00 AEST during winter — acceptable since sync is not time-critical.
     *
     * Auth: Vercel sets `Authorization: Bearer ${CRON_SECRET}` automatically when CRON_SECRET env var is configured.
     *
     * Body: returns {success, results: SyncResult[]}. Status code 500 if orchestrator throws.
     */
    export const dynamic = 'force-dynamic'
    export const maxDuration = 300  // 5 minutes; matches sync-all/route.ts

    export async function GET(req: NextRequest) {
      const auth = req.headers.get('authorization')
      if (auth !== `Bearer ${process.env.CRON_SECRET}`) {
        return NextResponse.json({ error: 'Unauthorized' }, { status: 401 })
      }

      try {
        const results = await runSyncForAllBusinesses()
        const errored = results.filter(r => r.status === 'error').length
        const partial = results.filter(r => r.status === 'partial').length
        return NextResponse.json({
          success: true,
          totalBusinesses: results.length,
          successCount: results.filter(r => r.status === 'success').length,
          partialCount: partial,
          erroredCount: errored,
          results,
        })
      } catch (err: any) {
        Sentry.captureException(err, { tags: { invariant: 'cron_sync_all_xero' } })
        return NextResponse.json({ success: false, error: String(err?.message ?? err) }, { status: 500 })
      }
    }
    ```
