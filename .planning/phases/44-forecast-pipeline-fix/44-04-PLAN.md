<!-- Canonical FY window math -->

For a business with `fiscal_year_start: 7` (Jul-start, the AU default) and today = 2026-04-27:
- Current FY = 2026 (Jul 2025 – Jun 2026); current month = April 2026.
- Current FY base month for canonical query: April 2026 → `fromDate=2026-04-01, toDate=2026-04-30, periods=11, timeframe=MONTH` returns columns May 2025 – April 2026 (12 months, capped at FY start where applicable — so 10 months for current FY YTD if we're in April).
- Prior FY = 2025 (Jul 2024 – Jun 2025); base month = LAST month of prior FY = June 2025 → `fromDate=2025-06-01, toDate=2025-06-30, periods=11, timeframe=MONTH` returns columns Jul 2024 – Jun 2025 (12 months).
- The orchestrator expects 24 expected months of coverage (12 × 2) for full-history tenants.

    import { createSupabaseServiceClient } from '@/lib/supabase/server-client'   // ADJUST to project's actual service-role helper path
    import { resolveBusinessIds } from '@/lib/utils/resolve-business-ids'
    import { parsePLByMonth, computeCoverage } from './pl-by-month-parser'
    import { reconcilePL, parseFYTotalResponse } from './pl-reconciler'
    import { generateFiscalMonthKeys, getCurrentFiscalYear } from '@/lib/utils/fiscal-year-utils'
    import * as Sentry from '@sentry/nextjs'

    export type SyncResult = { /* per the SyncResult interface above */ }
    export type SyncOptions = { fyOverride?: number; tenantIdFilter?: string }

    /**
     * D-01/D-05/D-06/D-07/D-08/D-09/D-10 orchestrator.
     * Single entry point: every other route (sync-all, refresh-pl, cron) calls this.
     */
    export async function syncBusinessXeroPL(
      businessId: string,
      opts: SyncOptions = {}
    ): Promise<SyncResult> {
      const supabase = createSupabaseServiceClient()
      const ids = await resolveBusinessIds(supabase, businessId)
      const profileId = ids.profileId

      // 1. Acquire advisory lock — single-flight per business across containers.
      // Lock is released automatically when the calling transaction commits/aborts.
      // Note: pg_advisory_xact_lock requires a transaction; supabase.rpc runs each call in one.
      await supabase.rpc('acquire_xero_sync_lock', { p_business_id: profileId })

      // 2. Open sync_jobs row in 'running' state.
      const { data: jobRow } = await supabase.from('sync_jobs').insert({
        business_id: profileId, job_type: 'xero_pl_sync', status: 'running',
      }).select('id').single()
      const syncJobId = jobRow.id
      const startedAt = new Date()
      let xeroRequestCount = 0
      let rowsInserted = 0
      let rowsUpdated = 0
      const allDiscrepancies: Discrepancy[] = []

      try {
        // 3. Resolve FY windows (current YTD + prior FY).
        const fyStartMonth = await getFiscalYearStartMonth(supabase, profileId)  // 7 for AU default; helper reads business_profiles.fiscal_year_start
        const today = new Date()
        const currentFY = opts.fyOverride ?? getCurrentFiscalYear(today, fyStartMonth)
        const priorFY = currentFY - 1
        const fyWindows = [
          { fy: currentFY, baseMonth: getCurrentMonthBoundary(today),    expectedMonths: monthsElapsedInFY(today, fyStartMonth, currentFY) },
          { fy: priorFY,   baseMonth: getLastMonthOfFY(priorFY, fyStartMonth), expectedMonths: 12 },
        ]

        // 4. Iterate every active connected Xero org.
        const { data: connections } = await supabase
          .from('xero_connections')
          .select('id, tenant_id, tenant_name, business_id')
          .in('business_id', ids.all)
          .eq('is_active', true)

        const allParsedRows: ParsedPLRow[] = []
        const coveragePerTenant: CoverageRecord[] = []

        for (const conn of (connections ?? [])) {
          if (opts.tenantIdFilter && conn.tenant_id !== opts.tenantIdFilter) continue

          const tokens = await getValidAccessToken(profileId)  // refreshes if needed; returns access token usable for THIS connection's tenant
          // NOTE: token-manager today returns a single tenant_id; the orchestrator passes conn.tenant_id explicitly in the header

          for (const window of fyWindows) {
            // 4a. Fetch by-month report
            const byMonthUrl = `https://api.xero.com/api.xro/2.0/Reports/ProfitAndLoss?fromDate=${window.baseMonth.start}&toDate=${window.baseMonth.end}&periods=11&timeframe=MONTH&standardLayout=false&paymentsOnly=false`
            const byMonthResp = await fetch(byMonthUrl, {
              headers: { 'Authorization': `Bearer ${tokens.accessToken}`, 'xero-tenant-id': conn.tenant_id, 'Accept': 'application/json' }
            })
            xeroRequestCount++
            if (!byMonthResp.ok) throw new Error(`Xero PL-by-month ${byMonthResp.status} for tenant ${conn.tenant_id} fy ${window.fy}`)
            const byMonthJson = await byMonthResp.json()
            const monthlyRows = parsePLByMonth(byMonthJson)
              .map(r => ({ ...r, business_id: profileId, tenant_id: conn.tenant_id, fiscal_year: window.fy }))

            // 4b. Fetch single-period FY total for reconciliation
            const fyStart = `${window.fy - 1}-${String(fyStartMonth).padStart(2, '0')}-01`
            const fyEndMonth = fyStartMonth === 1 ? 12 : fyStartMonth - 1
            const fyEndYear = fyStartMonth === 1 ? window.fy - 1 : window.fy
            const fyEnd = lastDayOfMonth(fyEndYear, fyEndMonth)
            const fyTotalUrl = `https://api.xero.com/api.xro/2.0/Reports/ProfitAndLoss?fromDate=${fyStart}&toDate=${fyEnd}&standardLayout=false&paymentsOnly=false`
            const fyTotalResp = await fetch(fyTotalUrl, {
              headers: { 'Authorization': `Bearer ${tokens.accessToken}`, 'xero-tenant-id': conn.tenant_id, 'Accept': 'application/json' }
            })
            xeroRequestCount++
            if (!fyTotalResp.ok) throw new Error(`Xero FY-total ${fyTotalResp.status} for tenant ${conn.tenant_id} fy ${window.fy}`)
            const fyTotals = parseFYTotalResponse(await fyTotalResp.json())

            // 4c. Reconcile (D-08 fail loud)
            const recResult = reconcilePL(monthlyRows, fyTotals, 0.01)
            if (recResult.status === 'mismatch') {
              allDiscrepancies.push(...recResult.discrepancies.map(d => ({ ...d, tenant_id: conn.tenant_id, fiscal_year: window.fy })))
            }

            // 4d. Upsert long-format rows (idempotent ON CONFLICT)
            const dbRows = monthlyRows.map(r => ({
              business_id: profileId,
              tenant_id: conn.tenant_id,
              account_code: r.account_code,
              account_name: r.account_name,
              account_type: r.account_type,
              period_month: r.period_month,
              amount: r.amount,
              fiscal_year: window.fy,
              source: 'xero',
              updated_at: new Date().toISOString(),
            }))
            const { error: upsertErr } = await supabase
              .from('xero_pl_lines')
              .upsert(dbRows, { onConflict: 'business_id,tenant_id,account_code,period_month', ignoreDuplicates: false })
            if (upsertErr) throw new Error(`xero_pl_lines upsert: ${upsertErr.message}`)
            rowsInserted += dbRows.length     // Supabase doesn't distinguish insert vs update in upsert; treat all as inserted+updated
            allParsedRows.push(...monthlyRows)
            coveragePerTenant.push(computeCoverage(monthlyRows, window.expectedMonths))
          }
        }

        // 5. Aggregate coverage across tenants for the sync_jobs row.
        const aggregateCoverage = aggregateCoverageRecords(coveragePerTenant)

        // 6. Update sync_jobs row to final status.
        const finalStatus: 'success' | 'partial' = allDiscrepancies.length === 0 ? 'success' : 'partial'
        await supabase.from('sync_jobs').update({
          status: finalStatus, finished_at: new Date().toISOString(),
          fy_range: { current_fy: currentFY, prior_fy: priorFY, fy_start_month: fyStartMonth },
          coverage: aggregateCoverage,
          rows_inserted: rowsInserted, rows_updated: rowsUpdated,
          xero_request_count: xeroRequestCount,
          reconciliation: allDiscrepancies.length > 0 ? { status: 'mismatch', discrepancies: allDiscrepancies } : { status: 'ok' },
          error: allDiscrepancies.length > 0 ? `Reconciliation mismatch on ${allDiscrepancies.length} accounts` : null,
        }).eq('id', syncJobId)

        if (allDiscrepancies.length > 0) {
          Sentry.captureMessage('xero_sync_reconciliation_mismatch', { tags: { invariant: 'reconciliation', business_id: profileId, sync_job_id: syncJobId } })
        }

        return {
          business_id: businessId, status: finalStatus, sync_job_id: syncJobId,
          rows_inserted: rowsInserted, rows_updated: rowsUpdated,
          xero_request_count: xeroRequestCount, coverage: aggregateCoverage,
          reconciliation: { status: allDiscrepancies.length === 0 ? 'ok' : 'mismatch', discrepancy_count: allDiscrepancies.length },
        }
      } catch (err: any) {
        await supabase.from('sync_jobs').update({
          status: 'error', finished_at: new Date().toISOString(),
          xero_request_count: xeroRequestCount, error: String(err?.message ?? err),
        }).eq('id', syncJobId)
        Sentry.captureException(err, { tags: { invariant: 'xero_sync_orchestrator', business_id: profileId, sync_job_id: syncJobId } })
        throw err
      }
    }

    /**
     * Iterate ALL businesses with active xero_connections. Used by Vercel Cron (44-05).
     * Sequential — concurrency limit is 1 — to keep within Vercel function maxDuration and Xero rate limits.
     */
    export async function runSyncForAllBusinesses(): Promise<SyncResult[]> {
      const supabase = createSupabaseServiceClient()
      const { data: connections } = await supabase.from('xero_connections').select('business_id').eq('is_active', true)
      const uniqueBusinessIds = Array.from(new Set((connections ?? []).map(c => c.business_id)))
      const results: SyncResult[] = []
      for (const businessId of uniqueBusinessIds) {
        try { results.push(await syncBusinessXeroPL(businessId)) }
        catch (err) { results.push({ business_id: businessId, status: 'error', error: String(err), /* zero counters */ ...({} as any) }) }
      }
      return results
    }
