// Aggregate per-tenant rows back into a wide shape for UI consumers.
// xero_pl_lines is long-format: one row per (business_id, tenant_id, account_code, period_month).
// We group by (account_code, account_name, account_type, period_month) and SUM amount across all tenants for the business.

const { data } = await supabase
  .from('xero_pl_lines')
  .select('account_code, account_name, account_type, period_month, amount, tenant_id, fiscal_year')
  .in('business_id', ids.all)

// Then in JS:
//   group by (account_code, account_name, account_type) → emit one DTO per account
//   inside each group, group by period_month → sum amount across tenants → emit { 'YYYY-MM': total } as monthly_values JSONB

type MonthlyComposite = {
  forecast_id: string;
  fiscal_year: number;
  rows: Array<{
    account_code: string | null;
    account_name: string;
    account_type: 'revenue' | 'cogs' | 'opex' | 'other_income' | 'other_expense';

```typescript
import * as Sentry from '@sentry/nextjs'

function assertComputedAtIsFresh(
  assumptionsUpdatedAt: string | null,
  computedAt: string | null,
  forecastId: string,
): void {
  if (!assumptionsUpdatedAt) return  // No assumptions saved yet — nothing to compare; skip.
  if (!computedAt || new Date(computedAt) < new Date(assumptionsUpdatedAt)) {
    const err = new Error(
      `[ForecastReadService] INVARIANT VIOLATED: forecast_pl_lines.computed_at (${computedAt}) ` +
      `is older than forecast_assumptions.updated_at (${assumptionsUpdatedAt}) ` +
      `for forecast=${forecastId}. POST /api/forecast/${forecastId}/recompute to remediate.`
    )
    Sentry.captureException(err, { tags: { invariant: 'forecast_freshness', forecast_id: forecastId } })
    throw err
  }
}

function assertCoverageNonNegative(coverage: { months_covered: number }, forecastId: string): void {
  if (coverage.months_covered < 0) {
    const err = new Error(`[ForecastReadService] INVARIANT VIOLATED: coverage.months_covered=${coverage.months_covered} for forecast=${forecastId}`)
    Sentry.captureException(err, { tags: { invariant: 'coverage_non_negative', forecast_id: forecastId } })
    throw err
  }
}

function assertReconciliation(
  forecastMonthsSum: number,
  assumptionsTotal: number,
  forecastId: string,
  category: string,
): void {
  const tolerance = 0.01
  if (Math.abs(forecastMonthsSum - assumptionsTotal) > tolerance) {
    const err = new Error(
      `[ForecastReadService] INVARIANT VIOLATED: sum(forecast_pl_lines.${category}.forecast_months)=${forecastMonthsSum} ` +
      `differs from assumptions ${category} total=${assumptionsTotal} by ${Math.abs(forecastMonthsSum - assumptionsTotal)} ` +
      `for forecast=${forecastId}. Diff > tolerance ${tolerance}.`
    )
    Sentry.captureException(err, { tags: { invariant: 'forecast_reconciliation', forecast_id: forecastId, category } })
    throw err
  }
}

    import * as Sentry from '@sentry/nextjs'
    import { resolveBusinessIds } from '@/lib/utils/resolve-business-ids'
    import type { SupabaseClient } from '@supabase/supabase-js'

    export type AccountType = 'revenue' | 'cogs' | 'opex' | 'other_income' | 'other_expense';

    export type MonthlyCompositeRow = {
      account_code: string | null;
      account_name: string;
      account_type: AccountType;
      monthly_values: Record<string, number>;
    };

    export type ForecastRow = {
      account_code: string | null;
      account_name: string;
      category: string;
      forecast_months: Record<string, number>;
    };

    export type CoverageRecord = {
      months_covered: number;
      first_period: string | null;
      last_period: string | null;
      expected_months: number;
    };

    export type MonthlyComposite = {
      forecast_id: string;
      business_id: string;
      fiscal_year: number;
      rows: MonthlyCompositeRow[];
      forecast_rows: ForecastRow[];
      coverage: CoverageRecord;
      computed_at: string | null;
      assumptions_updated_at: string | null;
    };

    export type CategorySubtotals = {
      revenue: number;
      cogs: number;
      gross_profit: number;
      opex: number;
      net_profit: number;
      other_income: number;
      other_expense: number;
    };

    export type CashflowProjection = {
      forecast_id: string;
      monthly_net: Record<string, number>;
    };

    export class ForecastReadService {
      constructor(private supabase: SupabaseClient) {}

      /**
       * D-13 — wide-shaped DTO for wizard / monthly report.
       * Aggregates long-format xero_pl_lines rows per (account, period_month) across tenants.
       * Asserts D-18 invariants and throws on violation.
       */
      async getMonthlyComposite(forecastId: string): Promise<MonthlyComposite> {
        // 1. Load forecast row.
        const { data: forecast, error: fError } = await this.supabase
          .from('financial_forecasts')
          .select('id, business_id, fiscal_year, is_active')
          .eq('id', forecastId)
          .maybeSingle()
        if (fError || !forecast) throw new Error(`Forecast ${forecastId} not found`)

        // 2. Resolve dual IDs.
        const ids = await resolveBusinessIds(this.supabase, forecast.business_id)

        // 3. Load forecast_assumptions + forecast_pl_lines in parallel.
        const [assRes, plLinesRes, xeroRowsRes] = await Promise.all([
          this.supabase.from('forecast_assumptions').select('data, updated_at').eq('forecast_id', forecastId).maybeSingle(),
          this.supabase.from('forecast_pl_lines').select('account_code, account_name, category, forecast_months, computed_at').eq('forecast_id', forecastId),
          this.supabase.from('xero_pl_lines')
            .select('account_code, account_name, account_type, period_month, amount, tenant_id, fiscal_year')
            .in('business_id', ids.all)
            .eq('fiscal_year', forecast.fiscal_year)
        ])

        const assumptionsUpdatedAt: string | null = assRes.data?.updated_at ?? null
        const computedAt: string | null = (plLinesRes.data ?? []).reduce<string | null>((min, r) => {
          if (!min) return r.computed_at
          if (!r.computed_at) return min
          return new Date(r.computed_at) < new Date(min) ? r.computed_at : min
        }, null)

        // 4. D-18 invariant: computed_at >= assumptions_updated_at.
        this.assertComputedAtIsFresh(assumptionsUpdatedAt, computedAt, forecastId)

        // 5. Aggregate xero_pl_lines from long → wide.
        const xeroRows = xeroRowsRes.data ?? []
        const grouped = new Map<string, MonthlyCompositeRow>()  // key: account_code (or account_name fallback)
        for (const row of xeroRows) {
          const key = row.account_code ?? row.account_name
          let agg = grouped.get(key)
          if (!agg) {
            agg = { account_code: row.account_code, account_name: row.account_name, account_type: row.account_type as AccountType, monthly_values: {} }
            grouped.set(key, agg)
          }
          const monthKey = (row.period_month as string).slice(0, 7)  // 'YYYY-MM' from 'YYYY-MM-01'
          agg.monthly_values[monthKey] = (agg.monthly_values[monthKey] ?? 0) + Number(row.amount)
        }

        // 6. Compute coverage.
        const allMonths = new Set<string>()
        for (const r of grouped.values()) for (const m of Object.keys(r.monthly_values)) allMonths.add(m)
        const sortedMonths = [...allMonths].sort()
        const coverage: CoverageRecord = {
          months_covered: sortedMonths.length,
          first_period: sortedMonths[0] ?? null,
          last_period: sortedMonths.at(-1) ?? null,
          expected_months: 12,  // CALLER overrides via opts; default = single-FY view
        }
        this.assertCoverageNonNegative(coverage, forecastId)

        const forecastRows: ForecastRow[] = (plLinesRes.data ?? []).map(r => ({
          account_code: r.account_code, account_name: r.account_name, category: r.category,
          forecast_months: (r.forecast_months as Record<string, number>) ?? {},
        }))

        return {
          forecast_id: forecastId,
          business_id: forecast.business_id,
          fiscal_year: forecast.fiscal_year,
          rows: [...grouped.values()],
          forecast_rows: forecastRows,
          coverage,
          computed_at: computedAt,
          assumptions_updated_at: assumptionsUpdatedAt,
        }
      }

      async getCategorySubtotalsForMonth(forecastId: string, month: string): Promise<CategorySubtotals> {
        const composite = await this.getMonthlyComposite(forecastId)
        const sumByType = (type: AccountType) => composite.rows
          .filter(r => r.account_type === type)
          .reduce((s, r) => s + (r.monthly_values[month] ?? 0), 0)
        const revenue = sumByType('revenue')
        const cogs = sumByType('cogs')
        const opex = sumByType('opex')
        const other_income = sumByType('other_income')
        const other_expense = sumByType('other_expense')
        return {
          revenue, cogs, gross_profit: revenue - cogs,
          opex, net_profit: revenue - cogs - opex + other_income - other_expense,
          other_income, other_expense,
        }
      }

      async getCashflowProjection(forecastId: string): Promise<CashflowProjection> {
        const composite = await this.getMonthlyComposite(forecastId)
        const monthly_net: Record<string, number> = {}
        for (const fRow of composite.forecast_rows) {
          for (const [m, v] of Object.entries(fRow.forecast_months)) {
            const sign = ['revenue', 'other_income'].includes(fRow.category) ? +1 : -1
            monthly_net[m] = (monthly_net[m] ?? 0) + sign * Number(v)
          }
        }
        return { forecast_id: forecastId, monthly_net }
      }

      // D-18 invariant assertions:
      private assertComputedAtIsFresh(assumptionsUpdatedAt: string | null, computedAt: string | null, forecastId: string): void {
        if (!assumptionsUpdatedAt) return
        if (!computedAt || new Date(computedAt) < new Date(assumptionsUpdatedAt)) {
          const err = new Error(`[ForecastReadService] INVARIANT VIOLATED: forecast_pl_lines.computed_at (${computedAt}) is older than forecast_assumptions.updated_at (${assumptionsUpdatedAt}) for forecast=${forecastId}. POST /api/forecast/${forecastId}/recompute to remediate.`)
          Sentry.captureException(err, { tags: { invariant: 'forecast_freshness', forecast_id: forecastId } })
          throw err
        }
      }

      private assertCoverageNonNegative(coverage: CoverageRecord, forecastId: string): void {
        if (coverage.months_covered < 0) {
          const err = new Error(`[ForecastReadService] INVARIANT VIOLATED: coverage.months_covered=${coverage.months_covered} for forecast=${forecastId}`)
          Sentry.captureException(err, { tags: { invariant: 'coverage_non_negative', forecast_id: forecastId } })
          throw err
        }
      }
    }

    export function createForecastReadService(supabase: SupabaseClient): ForecastReadService {
      return new ForecastReadService(supabase)
    }
    ```
