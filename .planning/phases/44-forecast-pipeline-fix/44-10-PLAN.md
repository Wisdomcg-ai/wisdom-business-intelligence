    ```tsx
    import { Database } from 'lucide-react'

    export type XeroCoverageBannerProps = {
      coverage: {
        monthsCovered: number;
        firstPeriod: string | null;       // 'YYYY-MM'
        lastPeriod: string | null;
        expectedMonths: number;
      };
    }

    const MONTH_NAMES = ['Jan', 'Feb', 'Mar', 'Apr', 'May', 'Jun', 'Jul', 'Aug', 'Sep', 'Oct', 'Nov', 'Dec'] as const

    function fmtMonYYYY(yyyymm: string | null): string {
      if (!yyyymm) return ''
      const [y, m] = yyyymm.split('-')
      const idx = parseInt(m, 10) - 1
      if (idx < 0 || idx > 11) return yyyymm
      return `${MONTH_NAMES[idx]} ${y}`
    }

    /**
     * Phase 44 D-15 — Xero coverage banner.
     *
     * Visibility: only renders when coverage.monthsCovered < coverage.expectedMonths.
     * Non-dismissible. Non-interactive. role="status" so screen readers announce on mount.
     *
     * Visual contract (binding) — see .planning/phases/44-forecast-pipeline-fix/44-UI-SPEC.md:
     *   bg-amber-50 border border-amber-200 rounded-lg p-4 mb-4
     *   Database icon (lucide-react), text-amber-700, w-5 h-5
     *   Eyebrow: text-xs font-bold uppercase tracking-wide text-amber-700 mb-1
     *   Main line: text-sm font-semibold text-brand-navy
     *   Sub-line: text-xs text-gray-600 mt-1 leading-relaxed
     *
     * Copy (binding):
     *   Eyebrow: 'XERO DATA COVERAGE'
     *   Main line (multi-month): 'Xero data covers {Mon YYYY} – {Mon YYYY} ({N} months)'
     *   Main line (single month): 'Xero data covers {Mon YYYY} only (1 month)'
     *   Sub-line: 'Forecast defaults are calculated from these months only. You can override any value below.'
     */
    export function XeroCoverageBanner({ coverage }: XeroCoverageBannerProps) {
      const { monthsCovered, firstPeriod, lastPeriod, expectedMonths } = coverage

      // Visibility rule: only show when coverage is partial.
      if (monthsCovered >= expectedMonths) return null
      if (!firstPeriod || !lastPeriod) return null

      const first = fmtMonYYYY(firstPeriod)
      const last = fmtMonYYYY(lastPeriod)
      const mainLine =
        monthsCovered === 1
          ? `Xero data covers ${first} only (1 month)`
          : `Xero data covers ${first} – ${last} (${monthsCovered} months)`

      return (
        <div role="status" className="bg-amber-50 border border-amber-200 rounded-lg p-4 mb-4 flex items-start gap-3">
          <Database className="w-5 h-5 text-amber-700 mt-0.5 flex-shrink-0" aria-hidden="true" />
          <div className="flex-1">
            <p className="text-xs font-bold uppercase tracking-wide text-amber-700 mb-1">XERO DATA COVERAGE</p>
            <p className="text-sm font-semibold text-brand-navy">{mainLine}</p>
            <p className="text-xs text-gray-600 mt-1 leading-relaxed">
              Forecast defaults are calculated from these months only. You can override any value below.
            </p>
          </div>
        </div>
      )
    }
    ```
