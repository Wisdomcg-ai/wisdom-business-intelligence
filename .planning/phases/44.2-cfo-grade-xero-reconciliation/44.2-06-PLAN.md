    // D-44.2-08 absorber — compute per-account FY-vs-monthly residual.
    const monthlyRows = parsePLByMonth(byMonthJson)
    const fyTotals = parseFYTotalResponse(fyTotalJson)

    // Group monthlyRows by account-key, compute monthly_sum per group.
    const sumsByKey: Record<string, { rows: ParsedPLRow[]; sum: number; lastRow: ParsedPLRow }> = {}
    for (const row of monthlyRows) {
      const key = row.account_code ?? `NAME:${row.account_name}`
      const g = sumsByKey[key] ?? { rows: [], sum: 0, lastRow: row }
      g.rows.push(row)
      g.sum += row.amount
      g.lastRow = row
      sumsByKey[key] = g
    }

    // Inject synthetic adjustment rows for any account where |diff| > 0.01.
    const fyEndMonth = `${window.fy}-${String(fyStartMonth === 7 ? 6 : 12).padStart(2, '0')}-01`
    const adjustments: ParsedPLRow[] = []
    for (const [key, g] of Object.entries(sumsByKey)) {
      const fyTotal = fyTotals[key] ?? 0
      const diff = fyTotal - g.sum
      if (Math.abs(diff) > 0.01) {
        adjustments.push({
          account_code: g.lastRow.account_code,
          account_name: g.lastRow.account_name,
          account_type: g.lastRow.account_type,
          period_month: fyEndMonth,
          amount: diff,
          notes: `YE-ADJ: ${diff.toFixed(2)} absorbed via D-44.2-08; FY total trumps by-month sum`,
        })
      }
    }
    const adjustedRows = [...monthlyRows, ...adjustments]
    const recResult = reconcilePL(adjustedRows, fyTotals, 0.01)
