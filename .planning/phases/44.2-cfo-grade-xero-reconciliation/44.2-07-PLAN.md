type DataQuality =
  | 'verified'    // most recent sync for tenant: status='success'
  | 'partial'     // most recent sync: status='partial' (reconciler caught discrepancies)
  | 'failed'      // most recent sync: status='failed' (sync threw, no data)
  | 'no_sync'     // no sync_jobs row for this (business_id, tenant_id)
  | 'stale'       // last successful sync >24h ago + no newer attempt
```

D-44.2-04: per-tenant first class. Worst-of-tenants ordering:
  failed > partial > no_sync > stale > verified (severity desc)

sync_jobs schema (post-44.2-02):
- (business_id, tenant_id, started_at) → uniqueness via index sync_jobs_business_tenant_started_idx
- status: 'running' | 'success' | 'partial' | 'error' (note: 'error' not 'failed' — see existing migration line 25)

Mapping sync_jobs.status → DataQuality:
- 'success' → 'verified'
- 'partial' → 'partial'
- 'error' → 'failed'
- 'running' → use the previous final-status row if it exists, else 'no_sync'
- (no row) → 'no_sync'
- 'success' but >24h old AND no newer attempt → 'stale'

    /**
     * D-44.2-03 / D-44.2-04 — per-tenant data_quality, rolled up to
     * business level via worst-of severity.
     *
     * Reads sync_jobs LATEST row per (business_id, tenant_id) using the
     * sync_jobs_business_tenant_started_idx index from 44.2-02.
     *
     * @returns business-level data_quality + per-tenant breakdown
     */
    private async computeDataQuality(
      businessIds: string[],
    ): Promise<{ data_quality: DataQuality; per_tenant_quality: PerTenantQuality[] }> {
      // 1. Get distinct tenant_ids for these business_ids from xero_connections.
      //    sync_jobs alone won't surface tenants with no sync history yet.
      const { data: connections } = await this.supabase
        .from('xero_connections')
        .select('tenant_id, business_id')
        .in('business_id', businessIds)
        .eq('is_active', true)

      const tenants = new Set<string>()
      for (const c of (connections ?? [])) {
        if (c.tenant_id) tenants.add(c.tenant_id as string)
      }

      // 2. For EACH tenant, find the latest sync_jobs row.
      const perTenant: PerTenantQuality[] = []
      const STALE_THRESHOLD_MS = 24 * 60 * 60 * 1000  // 24h
      const now = Date.now()

      for (const tenantId of tenants) {
        const { data: latest } = await this.supabase
          .from('sync_jobs')
          .select('status, started_at, finished_at, reconciliation')
          .in('business_id', businessIds)
          .eq('tenant_id', tenantId)
          .order('started_at', { ascending: false })
          .limit(1)
          .maybeSingle()

        let quality: DataQuality
        let lastSyncAt: string | null = null
        let lastSyncStatus: string | null = null
        let discrepancyCount = 0

        if (!latest) {
          quality = 'no_sync'
        } else {
          lastSyncAt = (latest.started_at ?? null) as string | null
          lastSyncStatus = (latest.status ?? null) as string | null
          discrepancyCount = ((latest.reconciliation as any)?.discrepant_accounts?.length ?? 0) as number

          if (latest.status === 'success') {
            const ageMs = lastSyncAt ? now - new Date(lastSyncAt).getTime() : Infinity
            quality = ageMs > STALE_THRESHOLD_MS ? 'stale' : 'verified'
          } else if (latest.status === 'partial') {
            quality = 'partial'
          } else if (latest.status === 'error') {
            quality = 'failed'
          } else {
            // 'running' or unknown — treat as no_sync (no completed result)
            quality = 'no_sync'
          }
        }

        perTenant.push({
          tenant_id: tenantId,
          data_quality: quality,
          last_sync_at: lastSyncAt,
          last_sync_status: lastSyncStatus,
          discrepancy_count: discrepancyCount,
        })
      }

      // 3. Worst-of severity across tenants (D-44.2-04).
      const severity: Record<DataQuality, number> = {
        verified: 0,
        stale: 1,
        no_sync: 2,
        partial: 3,
        failed: 4,
      }
      let business: DataQuality = perTenant.length === 0 ? 'no_sync' : 'verified'
      for (const t of perTenant) {
        if (severity[t.data_quality] > severity[business]) {
          business = t.data_quality
        }
      }

      return { data_quality: business, per_tenant_quality: perTenant }
    }
    ```
