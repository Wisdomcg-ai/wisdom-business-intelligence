<objective>
Belt-and-braces continuous integrity check (D-44.2-15, D-44.2-16). The on-sync reconciler runs only when sync runs (could be hours apart). The daily cron catches drift between syncs by directly querying Xero's P&L API for each (business_id, tenant_id, current FY) and running the reconciler.

On mismatch: writes a sync_jobs row with status='partial' (so the existing data_quality gate from 44.2-07 picks it up automatically) AND tags Sentry with `continuous_reconciliation_drift` so operators see drift events even when no user is on the platform.

This plan is autonomous=false because vercel.json cron registration + CRON_SECRET env var require user input (Vercel dashboard env config + verification of cron schedule).

Output: cron route + vercel.json entry + test.

    ```typescript
    /**
     * Phase 44.2 Plan 44.2-10 — Daily continuous reconciliation cron (D-44.2-15, D-44.2-16).
     *
     * Belt-and-braces vs the on-sync reconciler. Runs once a day, fetches
     * Xero P&L for the current FY for every active (business_id, tenant_id),
     * runs reconcilePL, writes a sync_jobs row with job_type='continuous_reconciliation'.
     *
     * On mismatch: Sentry tag 'continuous_reconciliation_drift' fires + sync_jobs.status='partial'.
     * The data_quality gate in ForecastReadService (44.2-07) auto-picks up the
     * 'partial' status for the next read; banner appears on the wizard.
     *
     * Auth: Bearer CRON_SECRET (Vercel cron pattern).
     */

    import { NextRequest, NextResponse } from 'next/server'
    import { createClient } from '@supabase/supabase-js'
    import * as Sentry from '@sentry/nextjs'
    import { getValidAccessToken } from '@/lib/xero/token-manager'
    import { parsePLByMonth } from '@/lib/xero/pl-by-month-parser'
    import { parseFYTotalResponse, reconcilePL } from '@/lib/xero/pl-reconciler'
    import {
      getCurrentFiscalYear,
      getFiscalYearStartDate,
      getFiscalYearEndDate,
      DEFAULT_YEAR_START_MONTH,
    } from '@/lib/utils/fiscal-year-utils'

    export const dynamic = 'force-dynamic'
    export const maxDuration = 300 // 5 min — cron may iterate many tenants

    export async function GET(request: NextRequest) {
      // Auth gate (Vercel cron pattern)
      const authHeader = request.headers.get('authorization')
      if (authHeader !== `Bearer ${process.env.CRON_SECRET}`) {
        return new NextResponse('Unauthorized', { status: 401 })
      }

      const supabase = createClient(
        process.env.NEXT_PUBLIC_SUPABASE_URL!,
        process.env.SUPABASE_SERVICE_ROLE_KEY ?? process.env.SUPABASE_SERVICE_KEY!,
      )

      // 1. Iterate ALL active xero_connections.
      const { data: connections, error: connErr } = await supabase
        .from('xero_connections')
        .select('id, business_id, tenant_id, tenant_name')
        .eq('is_active', true)

      if (connErr) {
        Sentry.captureException(connErr, { tags: { cron: 'daily-reconciliation' } })
        return NextResponse.json({ error: 'connections fetch failed' }, { status: 500 })
      }

      const yearStartMonth = DEFAULT_YEAR_START_MONTH
      const fy = getCurrentFiscalYear(yearStartMonth)
      const fyStart = getFiscalYearStartDate(fy, yearStartMonth)
      const fyEnd = getFiscalYearEndDate(fy, yearStartMonth)
      const fmt = (d: Date) => `${d.getFullYear()}-${String(d.getMonth() + 1).padStart(2, '0')}-${String(d.getDate()).padStart(2, '0')}`
      const fyStartStr = fmt(fyStart)
      const fyEndStr = fmt(fyEnd)
      // Base month for by-month query: FY end month
      const baseFrom = fmt(new Date(fyEnd.getFullYear(), fyEnd.getMonth(), 1))
      const baseTo = fyEndStr

      const results: Array<{ tenant_id: string, status: string, discrepancies: number, error?: string }> = []

      // 2. Per-tenant: fetch Xero, parse, reconcile, write sync_jobs.
      let skippedRunningCount = 0
      for (const conn of (connections ?? [])) {
        try {
          // B5 — Running-sync guard. If sync-all-xero (or another job) is
          // currently running for this (business_id, tenant_id), SKIP this
          // pair to avoid (a) hammering the same Xero tenant in parallel and
          // (b) the sync orchestrator's xero_pl_lines writes racing with the
          // reconciler reading them. The next day's run picks up any pair we
          // skip today; one missed daily check is acceptable, a race is not.
          //
          // Schedule offset (`0 18 * * *`, 2h after sync-all-xero `0 16 * * *`)
          // is the primary defence; this guard is defence-in-depth in case a
          // sync runs long or someone manually triggers a sync near 18:00 UTC.
          const { data: runningRow } = await supabase
            .from('sync_jobs')
            .select('id, started_at')
            .eq('business_id', conn.business_id)
            .eq('tenant_id', conn.tenant_id)
            .eq('status', 'running')
            .order('started_at', { ascending: false })
            .limit(1)
            .maybeSingle()
          if (runningRow) {
            skippedRunningCount += 1
            results.push({
              tenant_id: conn.tenant_id,
              status: 'skipped_running',
              discrepancies: 0,
              error: `skipped: sync_jobs.status='running' since ${runningRow.started_at} for this (business_id, tenant_id) — next daily run will retry`,
            })
            console.log(`[cron/daily-reconciliation] SKIP business=${conn.business_id} tenant=${conn.tenant_id} — running sync since ${runningRow.started_at}`)
            continue
          }

          const tokenResult = await getValidAccessToken({ id: conn.id }, supabase as any)
          if (!tokenResult.success || !tokenResult.accessToken) {
            results.push({ tenant_id: conn.tenant_id, status: 'error', discrepancies: 0, error: tokenResult.error ?? 'token failure' })
            continue
          }
          const headers = {
            Authorization: `Bearer ${tokenResult.accessToken}`,
            'xero-tenant-id': conn.tenant_id,
            Accept: 'application/json',
          }

          // by-month
          const byMonthUrl = `https://api.xero.com/api.xro/2.0/Reports/ProfitAndLoss?fromDate=${baseFrom}&toDate=${baseTo}&periods=11&timeframe=MONTH&standardLayout=false&paymentsOnly=false`
          const byMonthRes = await fetch(byMonthUrl, { headers })
          if (!byMonthRes.ok) throw new Error(`Xero by-month ${byMonthRes.status}`)
          const byMonthJson = await byMonthRes.json()
          await new Promise(r => setTimeout(r, 300))

          // FY total
          const fyUrl = `https://api.xero.com/api.xro/2.0/Reports/ProfitAndLoss?fromDate=${fyStartStr}&toDate=${fyEndStr}&standardLayout=false&paymentsOnly=false`
          const fyRes = await fetch(fyUrl, { headers })
          if (!fyRes.ok) throw new Error(`Xero FY ${fyRes.status}`)
          const fyJson = await fyRes.json()
          await new Promise(r => setTimeout(r, 300))

          const monthlyRows = parsePLByMonth(byMonthJson)
          const fyTotals = parseFYTotalResponse(fyJson)
          const recResult = reconcilePL(monthlyRows, fyTotals, 0.01)

          const status: 'success' | 'partial' = recResult.status === 'ok' ? 'success' : 'partial'
          const reconciliation = recResult.status === 'ok'
            ? null
            : { tenant_id: conn.tenant_id, discrepant_accounts: recResult.discrepancies.map(d => ({ ...d, tenant_id: conn.tenant_id })) }

          // Write sync_jobs row tagged as continuous reconciliation (NOT a sync).
          await supabase.from('sync_jobs').insert({
            business_id: conn.business_id,
            tenant_id: conn.tenant_id,
            job_type: 'continuous_reconciliation',
            status,
            started_at: new Date().toISOString(),
            finished_at: new Date().toISOString(),
            fy_range: { current_fy: fy, fy_start_month: yearStartMonth },
            reconciliation,
            xero_request_count: 2,
            error: status === 'partial' ? `${recResult.discrepancies.length} accounts mismatched in continuous reconciliation` : null,
          })

          if (status === 'partial') {
            Sentry.captureMessage('Continuous reconciliation drift detected', {
              level: 'warning',
              tags: { continuous_reconciliation_drift: 'true', tenant_id: conn.tenant_id, business_id: conn.business_id },
              extra: { discrepancy_count: recResult.discrepancies.length, top_discrepancies: recResult.discrepancies.slice(0, 5) },
            })
          }

          results.push({ tenant_id: conn.tenant_id, status, discrepancies: recResult.status === 'ok' ? 0 : recResult.discrepancies.length })
        } catch (err: any) {
          // Don't abort — log and continue with next tenant
          Sentry.captureException(err, { tags: { cron: 'daily-reconciliation', tenant_id: conn.tenant_id } })
          results.push({ tenant_id: conn.tenant_id, status: 'error', discrepancies: 0, error: String(err?.message ?? err) })
        }
      }

      return NextResponse.json({
        ran_at: new Date().toISOString(),
        fiscal_year: fy,
        tenants_checked: results.length,
        partial_count: results.filter(r => r.status === 'partial').length,
        error_count: results.filter(r => r.status === 'error').length,
        skipped_running_count: skippedRunningCount,
        results,
      })
    }
    ```
