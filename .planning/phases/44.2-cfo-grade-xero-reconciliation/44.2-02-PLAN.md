<objective>
Close the multi-tenant audit-trail gap (D-44.2-05/06) by enforcing sync_jobs.tenant_id NOT NULL after backfill and by fixing the sync orchestrator to write tenant_id on every job. Without this, per-tenant reconciliation status (D-44.2-04) cannot exist — the wizard data_quality flag (44.2-07) needs to look up the LATEST sync per (business_id, tenant_id), and a NULL tenant_id makes that lookup impossible for consolidated entities.

Purpose: Multi-tenant first-class per D-44.2-04. The data_quality field gate in 44.2-07 reads `sync_jobs WHERE business_id=X AND tenant_id=Y ORDER BY started_at DESC LIMIT 1` — that query is a no-op when tenant_id is NULL.

Output: 1 migration + sync orchestrator fix + 1 regression test.
</objective>

    ```typescript
    for (const conn of activeConnections) {
      // INSERT sync_jobs row PER TENANT
      const { data: jobRow } = await supabase
        .from('sync_jobs')
        .insert({
          business_id: profileId,
          tenant_id: conn.tenant_id,  // D-44.2-05: NEVER null
          job_type: 'xero_pl_sync',
          status: 'running',
          started_at: new Date().toISOString(),
          fy_range: { current_fy: currentFY, prior_fy: priorFY, fy_start_month: fyStartMonth },
        })
        .select('id')
        .single();
      const tenantJobId = jobRow!.id;

      const tenantDiscrepancies: Array<Discrepancy & { tenant_id: string }> = [];
      const tenantCoveragePerWindow: CoverageRecord[] = [];
      let tenantRowsInserted = 0;
      let tenantXeroRequestCount = 0;

      for (const window of fyWindows) {
        // ... existing fetch + parse + reconcile + upsert logic ...

        if (recResult.status === 'mismatch') {
          for (const d of recResult.discrepancies) {
            // D-44.2-06: tenant_id in every discrepancy
            tenantDiscrepancies.push({ ...d, tenant_id: conn.tenant_id });
          }
        }
        tenantRowsInserted += dbRows.length;
        tenantXeroRequestCount += 2; // by-month + fy-total
        tenantCoveragePerWindow.push(computeCoverage(monthlyRows, window.expectedMonths));
      }

      // UPDATE sync_jobs row PER TENANT (per-tenant aggregation)
      const tenantExpectedTotal = fyWindows.reduce((sum, w) => sum + w.expectedMonths, 0);
      const tenantCoverage = aggregateCoverage(tenantCoveragePerWindow, tenantExpectedTotal);
      const tenantStatus: 'success' | 'partial' = tenantDiscrepancies.length === 0 ? 'success' : 'partial';
      const tenantError = tenantDiscrepancies.length > 0
        ? `Reconciliation mismatch on ${tenantDiscrepancies.length} accounts for tenant ${conn.tenant_id}`
        : null;

      await supabase.from('sync_jobs').update({
        status: tenantStatus,
        finished_at: new Date().toISOString(),
        coverage: tenantCoverage,
        reconciliation: tenantDiscrepancies.length > 0
          ? { tenant_id: conn.tenant_id, discrepant_accounts: tenantDiscrepancies }
          : null,
        rows_inserted: tenantRowsInserted,
        xero_request_count: tenantXeroRequestCount,
        error: tenantError,
      }).eq('id', tenantJobId);
    }
    ```
