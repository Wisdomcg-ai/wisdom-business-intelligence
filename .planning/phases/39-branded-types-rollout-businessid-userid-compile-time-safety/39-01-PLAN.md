  <files>src/lib/types/__tests__/ids.test-d.ts</files>
  <action>
    Create a new file that uses `// @ts-expect-error` directives to prove
    type-level enforcement. This file is NOT run as a runtime test — its
    success is measured by tsc passing (the expected errors occur, so
    @ts-expect-error is satisfied, which means tsc is happy).

    Contents:

    ```ts
    /**
     * Type-level test for branded ID types. Success == tsc exits 0.
     *
     * Each `@ts-expect-error` line claims "the next line IS a type error."
     * If we accidentally removed the brand, those lines would compile, the
     * directive would become unused, and tsc would emit "Unused
     * '@ts-expect-error' directive" — failing the build.
     */

    import type { BusinessId, UserId, BusinessProfileId } from '../ids'
    import { toBusinessId, toUserId, toBusinessProfileId } from '../ids'

    // Positive cases — the happy path
    const bid: BusinessId = toBusinessId('biz-uuid')
    const uid: UserId = toUserId('user-uuid')
    const pid: BusinessProfileId = toBusinessProfileId('profile-uuid')

    function needsBusiness(_id: BusinessId): void {}
    function needsUser(_id: UserId): void {}
    function needsProfile(_id: BusinessProfileId): void {}

    needsBusiness(bid)  // OK
    needsUser(uid)      // OK
    needsProfile(pid)   // OK

    // Negative cases — the compiler MUST reject these

    // @ts-expect-error raw string not assignable to BusinessId
    needsBusiness('raw-string')

    // @ts-expect-error UserId not assignable to BusinessId
    needsBusiness(uid)

    // @ts-expect-error BusinessId not assignable to UserId (brands are mutually exclusive)
    needsUser(bid)

    // @ts-expect-error BusinessProfileId not assignable to BusinessId
    needsBusiness(pid)

    // Branded IDs CAN be used where plain string is expected (subtype)
    function takesString(_s: string): void {}
    takesString(bid)  // OK — BusinessId extends string
    takesString(uid)  // OK

    export {}
    ```

    The filename uses `.test-d.ts` (a convention for "type-test only, not
    runtime") — just ensure it compiles as part of the project. If the
    existing vitest config picks it up, that's fine; no runtime assertions.
  </action>
  <acceptance_criteria>
    - File exists at src/lib/types/__tests__/ids.test-d.ts
    - `npx tsc --noEmit` passes (implying all @ts-expect-error directives are satisfied — the brand works)
    - If someone removes a brand accidentally, this file starts failing tsc
  </acceptance_criteria>
  <done>Compile-time regression test in place.</done>
