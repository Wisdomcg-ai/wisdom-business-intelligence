<objective>
Centralize all Xero refresh-token traffic through `src/lib/xero/token-manager.ts:getValidAccessToken`. Eliminate the two duplicate refresh implementations that bypass the lock and have over-eager 400-deactivation logic.

After this plan ships:
1. **Deleted:** `src/app/api/Xero/refresh-tokens/route.ts` — confirmed dead code (zero callers in src/, vercel.json, scripts, or FE). Its over-eager `is_active=false` flip on any HTTP 400 was the most likely root cause of the 2026-05-05 JDS connection drop (per 53-RESEARCH §4 worst-case interleaving).
2. **Refactored:** `src/app/api/Xero/reactivate/route.ts` — no longer calls `identity.xero.com/connect/token` directly. Calls `getValidAccessToken({ id: connection.id }, supabaseAdmin)` and on success flips `is_active=true`. Inherits 53-03's tightened deactivation policy (re-fetch row before deactivating, `unauthorized_client` retries 3×, etc.) for free.
3. **Verified:** the 16 existing `getValidAccessToken` consumers still work — `getValidAccessToken`'s public signature is unchanged by 53-03, so no source edits needed in any of: forecast/cashflow/{sync-balances, bank-balances, capex}, Xero/{chart-of-accounts-full, chart-of-accounts, balance-sheet, reconciliation, subscription-transactions, status, sync, employees}, monthly-report/{sync-xero, commentary, subscription-detail, wages-detail}, and sync-orchestrator.

**The point of 53-02** is not just code dedup — it's that the lock + retry + careful deactivation logic in token-manager only protects callers that ROUTE through it. Every parallel implementation re-introduced the rotation race that hit JDS. Eliminating duplicates closes that hole permanently.

```typescript
// src/lib/xero/token-manager.ts (verbatim public surface)

export type TokenRefreshError =
  | 'token_expired_permanently'
  | 'token_revoked'
  | 'network_error'
  | 'rate_limited'
  | 'server_error'
  | 'database_error'
  | 'unknown';

export interface TokenRefreshResult {
  success: boolean;
  accessToken?: string;
  error?: TokenRefreshError;
  message?: string;
  shouldDeactivate?: boolean;
}

export interface XeroConnection {
  id: string;
  business_id: string;
  tenant_id: string;
  tenant_name: string;
  access_token: string;
  refresh_token: string;
  expires_at: string;
  is_active: boolean;
  token_refreshing_at?: string | null;
}

/**
 * Get a valid access token, refreshing if needed.
 * Accepts EITHER a full XeroConnection OR a partial { id } object —
 * the function always re-fetches the row internally.
 *
 * Use { id } when the caller hasn't already loaded the row (preferred for
 * reactivate where we just need the id). Use full XeroConnection when the
 * caller already has the row in hand (saves one DB round trip — but
 * getValidAccessToken re-fetches anyway, so the "saving" is illusory; both
 * forms are accepted for ergonomic compatibility).
 */
export async function getValidAccessToken(
  connectionOrId: XeroConnection | { id: string },
  supabase: SupabaseClient
): Promise<TokenRefreshResult>;
```

## Reactivate route — refactored shape

```typescript
// src/app/api/Xero/reactivate/route.ts (post-53-02)

import { NextRequest, NextResponse } from 'next/server';
import { createClient } from '@supabase/supabase-js';
import { createRouteHandlerClient } from '@/lib/supabase/server';
import { resolveXeroBusinessId } from '@/lib/utils/resolve-xero-business-id';
import { getValidAccessToken } from '@/lib/xero/token-manager';

export const dynamic = 'force-dynamic';

const supabaseAdmin = createClient(
  process.env.NEXT_PUBLIC_SUPABASE_URL!,
  process.env.SUPABASE_SERVICE_KEY!
);

export async function POST(request: NextRequest) {
  try {
    // 1. Auth (UNCHANGED — owner / coach / super_admin)
    const supabase = await createRouteHandlerClient();
    const { data: { user }, error: userError } = await supabase.auth.getUser();
    if (userError || !user) return NextResponse.json({ error: 'Unauthorized' }, { status: 401 });

    const { business_id } = await request.json();
    if (!business_id) return NextResponse.json({ error: 'business_id is required' }, { status: 400 });

    const { data: business } = await supabaseAdmin
      .from('businesses').select('id, owner_id, assigned_coach_id').eq('id', business_id).single();
    if (!business) return NextResponse.json({ error: 'Business not found' }, { status: 404 });

    let allowed = business.owner_id === user.id || business.assigned_coach_id === user.id;
    if (!allowed) {
      const { data: roleRow } = await supabase
        .from('system_roles').select('role').eq('user_id', user.id).maybeSingle();
      if (roleRow?.role === 'super_admin') allowed = true;
    }
    if (!allowed) return NextResponse.json({ error: 'Access denied' }, { status: 403 });

    // 2. Resolve to canonical/profile ID + fetch connection (UNCHANGED)
    const { connectionBusinessId } = await resolveXeroBusinessId(supabaseAdmin, business_id);
    const { data: connection, error: connError } = await supabaseAdmin
      .from('xero_connections').select('*')
      .eq('business_id', connectionBusinessId)
      .order('updated_at', { ascending: false }).limit(1).single();

    if (connError || !connection) {
      return NextResponse.json({
        success: false, error: 'no_connection',
        message: 'No Xero connection found. Please connect Xero from the Integrations page.'
      }, { status: 404 });
    }

    if (connection.is_active) {
      return NextResponse.json({ success: true, message: 'Connection is already active', was_inactive: false });
    }

    // 3. NEW: delegate refresh to centralized token-manager.
    //    Note: token-manager fetches the row itself and runs the lock + retry + deactivation policy
    //    from 53-03. We do NOT decrypt the refresh_token here — token-manager owns that.
    //    Pass { id } so the function re-fetches a fresh row (avoids stale-in-memory rotation race).
    const tokenResult = await getValidAccessToken({ id: connection.id }, supabaseAdmin);

    if (!tokenResult.success) {
      // token-manager has already classified the error and (if shouldDeactivate) flipped is_active=false.
      // Map its error categories to the same response shape this route used pre-refactor so any FE
      // logic that branches on `error: 'token_expired'` still works.
      if (tokenResult.error === 'token_expired_permanently') {
        return NextResponse.json({
          success: false, error: 'token_expired',
          message: 'Refresh token has expired. Please reconnect Xero from the Integrations page.'
        }, { status: 401 });
      }
      if (tokenResult.error === 'token_revoked') {
        return NextResponse.json({
          success: false, error: 'token_expired',
          message: 'Access has been revoked. Please reconnect Xero from the Integrations page.'
        }, { status: 401 });
      }
      // Transient — token-manager already retried internally. Surface as 500 with detail.
      return NextResponse.json({
        success: false, error: 'refresh_failed',
        message: tokenResult.message ?? 'Token refresh failed'
      }, { status: 500 });
    }

    // 4. Refresh succeeded — flip is_active=true. token-manager already saved fresh tokens
    //    + new expires_at. We only need the activation flag flip.
    const { error: updateError } = await supabaseAdmin
      .from('xero_connections')
      .update({ is_active: true, updated_at: new Date().toISOString() })
      .eq('id', connection.id);

    if (updateError) {
      return NextResponse.json({
        success: false, error: 'save_failed',
        message: 'Token refreshed but failed to flip is_active=true'
      }, { status: 500 });
    }

    // 5. Re-read updated row to return the fresh expires_at to the caller (FE shows "expires in N min")
    const { data: refreshedRow } = await supabaseAdmin
      .from('xero_connections')
      .select('id, tenant_name, expires_at')
      .eq('id', connection.id).single();

    return NextResponse.json({
      success: true,
      message: 'Xero connection has been reactivated',
      was_inactive: true,
      connection: {
        id: connection.id,
        tenant_name: connection.tenant_name,
        expires_at: refreshedRow?.expires_at ?? connection.expires_at
      }
    });

  } catch (error) {
    console.error('[Xero Reactivate] Error:', error);
    return NextResponse.json({
      success: false, error: 'internal_error',
      message: 'Failed to reactivate connection'
    }, { status: 500 });
  }
}
```
