## /api/Xero/connection-health route

```typescript
// src/app/api/Xero/connection-health/route.ts
import { NextRequest, NextResponse } from 'next/server';
import { createClient } from '@supabase/supabase-js';
import { createRouteHandlerClient } from '@/lib/supabase/server';

export const dynamic = 'force-dynamic';

const supabaseAdmin = createClient(
  process.env.NEXT_PUBLIC_SUPABASE_URL!,
  process.env.SUPABASE_SERVICE_KEY!
);

export type ConnectionHealthStatus = 'verified' | 'stale' | 'dead' | 'none';

export interface ConnectionHealthResult {
  business_id: string;
  status: ConnectionHealthStatus;
  last_refresh_at: string | null;
  expires_at: string | null;
  connection_id: string | null;
}

const VERIFIED_WINDOW_MS = 24 * 60 * 60 * 1000;       // 24h
const STALE_WINDOW_MS = 7 * 24 * 60 * 60 * 1000;       // 7d
const EXPIRES_GRACE_MS = 30 * 60 * 1000;               // 30min

export async function GET(request: NextRequest) {
  // 1. Auth
  const supabase = await createRouteHandlerClient();
  const { data: { user } } = await supabase.auth.getUser();
  if (!user) return NextResponse.json({ error: 'Unauthorized' }, { status: 401 });

  // 2. Parse business_ids[] (Next.js style: ?business_ids[]=a&business_ids[]=b)
  const url = new URL(request.url);
  const requested = url.searchParams.getAll('business_ids[]');
  if (requested.length === 0) {
    return NextResponse.json({ results: [] });
  }
  if (requested.length > 200) {
    return NextResponse.json({ error: 'Too many business_ids' }, { status: 400 });
  }

  // 3. RBAC filter — keep only business_ids the user can access (owner / coach / super_admin)
  const { data: roleRow } = await supabase
    .from('system_roles').select('role').eq('user_id', user.id).maybeSingle();
  const isSuperAdmin = roleRow?.role === 'super_admin';

  let allowedIds: string[] = [];
  if (isSuperAdmin) {
    allowedIds = requested;
  } else {
    const { data: businesses } = await supabaseAdmin
      .from('businesses')
      .select('id, owner_id, assigned_coach_id')
      .in('id', requested);
    allowedIds = (businesses ?? [])
      .filter(b => b.owner_id === user.id || b.assigned_coach_id === user.id)
      .map(b => b.id);
  }

  if (allowedIds.length === 0) return NextResponse.json({ results: [] });

  // 4. Dual-ID expansion — also include business_profiles.id forms so legacy rows show up
  const { data: profiles } = await supabaseAdmin
    .from('business_profiles')
    .select('id, business_id')
    .in('business_id', allowedIds);
  const profileIdToBizId = new Map<string, string>();
  for (const p of (profiles ?? [])) profileIdToBizId.set(p.id, p.business_id);
  const allIdForms = [...allowedIds, ...(profiles ?? []).map(p => p.id)];

  // 5. Single batched query for all relevant connections
  const { data: connections } = await supabaseAdmin
    .from('xero_connections')
    .select('id, business_id, is_active, last_synced_at, updated_at, expires_at')
    .in('business_id', allIdForms)
    .order('updated_at', { ascending: false });   // most-recent per business wins

  // 6. Bucket by canonical business_id; pick the most recent active row, else the most recent dead row
  const byBizId = new Map<string, typeof connections[0] | null>();
  for (const id of allowedIds) byBizId.set(id, null);
  for (const conn of (connections ?? [])) {
    const canonicalId = profileIdToBizId.get(conn.business_id) ?? conn.business_id;
    const existing = byBizId.get(canonicalId);
    if (!existing) {
      byBizId.set(canonicalId, conn);
    } else if (!existing.is_active && conn.is_active) {
      // Prefer active over dead
      byBizId.set(canonicalId, conn);
    }
    // else keep existing (already prefer active OR same-status more-recent due to order)
  }

  // 7. Compute status per business
  const now = Date.now();
  const results: ConnectionHealthResult[] = allowedIds.map(business_id => {
    const conn = byBizId.get(business_id);
    if (!conn) {
      return { business_id, status: 'none', last_refresh_at: null, expires_at: null, connection_id: null };
    }
    if (!conn.is_active) {
      return { business_id, status: 'dead', last_refresh_at: conn.updated_at ?? null, expires_at: conn.expires_at ?? null, connection_id: conn.id };
    }
    const lastRefresh = Math.max(
      conn.last_synced_at ? new Date(conn.last_synced_at).getTime() : 0,
      conn.updated_at ? new Date(conn.updated_at).getTime() : 0,
    );
    const expiresAtMs = conn.expires_at ? new Date(conn.expires_at).getTime() : 0;
    const ageMs = now - lastRefresh;
    const isFresh = ageMs < VERIFIED_WINDOW_MS || expiresAtMs > now + EXPIRES_GRACE_MS;
    const status: ConnectionHealthStatus = isFresh
      ? 'verified'
      : ageMs < STALE_WINDOW_MS
        ? 'stale'
        : 'stale';   // anything beyond 7d is still 'stale' until is_active=false flips
    // Note: we never return 'dead' for an is_active=true row even if last_refresh_at is ancient.
    // That's the responsibility of the deactivation logic in token-manager (53-03). Health surface only reflects state.
    return {
      business_id,
      status,
      last_refresh_at: lastRefresh > 0 ? new Date(lastRefresh).toISOString() : null,
      expires_at: conn.expires_at ?? null,
      connection_id: conn.id,
    };
  });

  return NextResponse.json({ results });
}
```
