<objective>
Close the two race holes in `src/lib/xero/token-manager.ts` that cause healthy Xero connections to be falsely marked `is_active=false` when concurrent processes refresh the same row, and tighten error categorization so transient credential errors don't permanently kill connections on first occurrence.

This is the policy-correctness plan within Phase 53. After this lands:
1. **Hole A closed** — `getValidAccessToken` re-fetches the connection row AFTER acquiring the lock and re-decrypts the refresh_token, eliminating the "stale in-memory refresh_token after sibling rotation" failure mode (53-RESEARCH.md §4 worst-case interleaving).
2. **Hole B closed** — `refreshTokenWithRetry` re-fetches the row BEFORE every deactivation write. If a sibling rotated successfully (expires_at moved past threshold OR updated_at advanced), the deactivation is suppressed and a transient/race-detected result is returned instead.
3. **`unauthorized_client` no longer deactivates on first occurrence** — gets 3 retries with exponential backoff per Nango writeup confirmation that Xero returns this code transiently during their internal client-config changes (53-RESEARCH.md §8).
4. **Structured deactivation logging** — every is_active=false write is preceded by a single console line carrying full Xero response context + decision rationale. This is the data scaffolding 53-05 will wrap with `Sentry.captureException`.

**Out of scope:**
- Sentry capture (53-05).
- Deleting `/api/Xero/refresh-tokens` duplicate (53-02).
- New disconnect endpoint (53-01).
- Cron route (53-04).
- UI changes (53-05).
- Any changes to the 16 `getValidAccessToken` call sites (53-RESEARCH.md §2 confirms signature compat).

**Output:** 1 modified library file (~80-120 net LOC added including helpers + comments), 1 new vitest test file (~350-450 LOC), 2 atomic commits (RED test + GREEN implementation; optional 3rd refactor commit if helpers warrant a clean-up pass).

```typescript
/**
 * Re-fetch a connection row after a token refresh failure to detect
 * whether a sibling process rotated the token while we were calling Xero.
 * Returns the latest row + a `raceDetected` flag (true if expires_at advanced
 * past the threshold OR updated_at is newer than the snapshot we held).
 */
async function refetchConnectionForRaceCheck(
  connectionId: string,
  preFailureUpdatedAt: string | undefined,
  supabase: SupabaseClient
): Promise<{
  row: XeroConnection | null;
  raceDetected: boolean;
}> {
  const { data, error } = await supabase
    .from('xero_connections')
    .select('id, expires_at, updated_at, access_token, is_active')
    .eq('id', connectionId)
    .single();
  if (error || !data) return { row: null, raceDetected: false };
  const now = new Date();
  const thresholdTime = new Date(now.getTime() + REFRESH_THRESHOLD_MINUTES * 60 * 1000);
  const expiresAtAdvanced = new Date(data.expires_at) > thresholdTime;
  const updatedAtAdvanced = preFailureUpdatedAt
    ? new Date(data.updated_at) > new Date(preFailureUpdatedAt)
    : false;
  return {
    row: data as XeroConnection,
    raceDetected: expiresAtAdvanced || updatedAtAdvanced,
  };
}
```

```typescript
// Pseudocode of the post-53-03 flow. Lines marked NEW are the changes.
export async function getValidAccessToken(connectionOrId, supabase) {
  const connectionId = connectionOrId.id;

  // Step 1: Initial fetch (existing)
  const { data: initialRow, error: fetchError } = await supabase
    .from('xero_connections')
    .select('*')
    .eq('id', connectionId)
    .single();
  if (fetchError || !initialRow) return { success: false, error: 'database_error', shouldDeactivate: false };

  // Step 2: Decrypt initial tokens (existing — but we may DISCARD these if a sibling rotated)
  let decryptedAccessToken = decrypt(initialRow.access_token);
  let decryptedRefreshToken = decrypt(initialRow.refresh_token);

  // Step 3: If still valid, short-circuit (existing)
  const now = new Date();
  const thresholdTime = new Date(now.getTime() + REFRESH_THRESHOLD_MINUTES * 60 * 1000);
  if (new Date(initialRow.expires_at) > thresholdTime) {
    return { success: true, accessToken: decryptedAccessToken };
  }

  // Step 4: Try to acquire the lock (existing)
  const lockResult = await acquireRefreshLock(connectionId, supabase);
  if (!lockResult.acquired) {
    // Existing wait-and-retry branch (preserved)
    await sleep(2000);
    const { data: updatedRow } = await supabase
      .from('xero_connections')
      .select('*')
      .eq('id', connectionId)
      .single();
    if (updatedRow && new Date(updatedRow.expires_at) > thresholdTime) {
      return { success: true, accessToken: decrypt(updatedRow.access_token) };
    }
    // If still expired, fall through and try ourselves (no lock — best-effort).
    // (53-03 does NOT change this branch — it's already race-safe via re-fetch.)
  }

  // ─── Step 5 NEW: Lock acquired → re-fetch row before calling Xero. ───────
  // This closes Hole A (53-RESEARCH.md §4) — sibling may have rotated between
  // steps 1 and 4. Use the FRESH row's refresh_token, not the stale snapshot.
  let postLockRow = initialRow;
  if (lockResult.acquired) {
    const { data: freshRow, error: refetchErr } = await supabase
      .from('xero_connections')
      .select('*')
      .eq('id', connectionId)
      .single();
    if (refetchErr || !freshRow) {
      await releaseRefreshLock(connectionId, supabase);
      return { success: false, error: 'database_error', shouldDeactivate: false };
    }
    postLockRow = freshRow;

    // If a sibling already refreshed during our lock-acquisition window,
    // expires_at will be past the threshold — short-circuit success.
    if (new Date(postLockRow.expires_at) > thresholdTime) {
      await releaseRefreshLock(connectionId, supabase);
      return { success: true, accessToken: decrypt(postLockRow.access_token) };
    }

    // Re-decrypt with the fresh refresh_token before calling Xero
    decryptedRefreshToken = decrypt(postLockRow.refresh_token);
  }

  // Step 6: Refresh with retry (existing — but `decryptedRefreshToken` is now fresh)
  try {
    const result = await refreshTokenWithRetry(
      decryptedRefreshToken,
      connectionId,
      supabase,
      postLockRow.updated_at,  // NEW: pass pre-refresh updated_at snapshot for race detection
    );
    return result;
  } finally {
    await releaseRefreshLock(connectionId, supabase);
  }
}
```

## refreshTokenWithRetry — new signature + flow

```typescript
async function refreshTokenWithRetry(
  refreshToken: string,
  connectionId: string,
  supabase: SupabaseClient,
  preRefreshUpdatedAt: string | undefined,    // NEW
  attempt: number = 1,
  // NEW: track the previous attempt's error code so we can decide whether
  // to keep retrying unauthorized_client or escalate.
  previousErrorCode?: string,
): Promise<TokenRefreshResult> {
  // … existing fetch + ok-path unchanged …

  // Error path — categorize FIRST
  const errorText = await response.text();
  const errorInfo = categorizeError(response.status, errorText, attempt);

  // NEW: If categorize says "would deactivate", do the post-failure refetch
  // BEFORE writing is_active=false. This closes Hole B.
  if (errorInfo.shouldDeactivate) {
    const { row: postFailureRow, raceDetected } = await refetchConnectionForRaceCheck(
      connectionId,
      preRefreshUpdatedAt,
      supabase,
    );

    if (raceDetected && postFailureRow) {
      // Sibling rotated successfully while we were calling Xero. Use their
      // result; do NOT deactivate.
      logDeactivationDecision({
        decision: 'no_deactivate',
        rationale: 'race_detected_no_deactivate',
        connection_id: connectionId,
        attempt,
        xero_status: response.status,
        xero_error_code: extractErrorCode(errorText),
        xero_error_body: truncate(errorText, 500),
        expires_at_pre: 'unknown',  // pre-refetch we didn't read this — log empty
        expires_at_post: postFailureRow.expires_at,
        updated_at_pre: preRefreshUpdatedAt,
        updated_at_post: postFailureRow.updated_at,
      });
      try {
        return { success: true, accessToken: decrypt(postFailureRow.access_token) };
      } catch {
        // Decrypt of sibling's access_token failed — return transient
        return { success: false, error: 'database_error', message: 'Race detected but failed to decrypt sibling token', shouldDeactivate: false };
      }
    }

    // Confirmed no race. Log + deactivate.
    logDeactivationDecision({
      decision: 'deactivate',
      rationale: errorInfo.message?.toLowerCase().includes('access has been revoked')
        ? 'access_denied_terminal'
        : extractErrorCode(errorText) === 'unauthorized_client'
          ? 'unauthorized_client_3x_exhausted'
          : 'invalid_grant_confirmed',
      connection_id: connectionId,
      attempt,
      xero_status: response.status,
      xero_error_code: extractErrorCode(errorText),
      xero_error_body: truncate(errorText, 500),
      expires_at_pre: 'unknown',
      expires_at_post: postFailureRow?.expires_at,
      updated_at_pre: preRefreshUpdatedAt,
      updated_at_post: postFailureRow?.updated_at,
    });

    await supabase
      .from('xero_connections')
      .update({ is_active: false, updated_at: new Date().toISOString() })
      .eq('id', connectionId);

    return errorInfo;
  }

  // Transient error path — retry with backoff (existing) …
  // BUT: if errorCode === 'unauthorized_client' AND attempt < MAX_RETRIES,
  // categorizeError returned shouldDeactivate=false → we retry. On the 3rd
  // attempt categorizeError returns shouldDeactivate=true (see new policy).
  if (attempt < MAX_RETRIES) {
    const delay = INITIAL_RETRY_DELAY_MS * Math.pow(2, attempt - 1);
    await sleep(delay);
    return refreshTokenWithRetry(refreshToken, connectionId, supabase, preRefreshUpdatedAt, attempt + 1, extractErrorCode(errorText));
  }

  // All retries exhausted on a transient code — return errorInfo (no deactivate)
  return errorInfo;
}
```

## categorizeError — new signature + per-attempt policy

```typescript
function categorizeError(
  status: number,
  errorText: string,
  attempt: number,            // NEW — needed to differentiate retry vs final unauthorized_client
): TokenRefreshResult {
  let errorData: any = {};
  try { errorData = JSON.parse(errorText); } catch {}
  const errorCode = errorData.error || '';

  // invalid_grant — terminal, deactivate (caller does the post-failure refetch)
  if (errorCode === 'invalid_grant') {
    return {
      success: false,
      error: 'token_expired_permanently',
      message: 'Refresh token has expired (60 days) or has been rotated. Please reconnect Xero.',
      shouldDeactivate: true,
    };
  }

  // access_denied — user revoked in Xero; terminal
  if (errorCode === 'access_denied') {
    return {
      success: false,
      error: 'token_revoked',
      message: 'Access has been revoked in Xero. Please reconnect.',
      shouldDeactivate: true,
    };
  }

  // unauthorized_client — Xero's transient client-credentials error per Nango
  // (53-RESEARCH.md §8). Retry on attempts 1-2; deactivate on 3rd.
  if (errorCode === 'unauthorized_client') {
    return {
      success: false,
      error: 'token_revoked',
      message:
        attempt < MAX_RETRIES
          ? 'Xero returned unauthorized_client (often transient); will retry.'
          : 'Xero returned unauthorized_client on 3 successive attempts.',
      shouldDeactivate: attempt >= MAX_RETRIES,
    };
  }

  // invalid_client — wrong client_id/secret. NEVER deactivate (ops/config bug).
  if (errorCode === 'invalid_client') {
    return {
      success: false,
      error: 'unknown',
      message: 'Xero rejected app credentials (invalid_client). This is an ops/config issue.',
      shouldDeactivate: false,
    };
  }

  // Generic 400 with no error field — preserve HEAD behavior (no deactivate)
  if (status === 400 && !errorCode) {
    return {
      success: false,
      error: 'unknown',
      message: `Bad request to Xero: ${errorText.slice(0, 200)}`,
      shouldDeactivate: false,
    };
  }

  // 429
  if (status === 429) {
    return { success: false, error: 'rate_limited', message: 'Too many requests to Xero.' };
  }

  // 5xx — transient
  if (status >= 500) {
    return { success: false, error: 'server_error', message: 'Xero API is temporarily unavailable.' };
  }

  // Catch-all — transient, no deactivate
  return {
    success: false,
    error: 'unknown',
    message: `Unexpected error: ${status} - ${errorText.slice(0, 200)}`,
  };
}
```

```typescript
// Build a chainable Supabase stub. State lives in a closure so tests can
// mutate the row mid-call to simulate sibling rotations.
function makeMockSupabase(initialRow: XeroConnection) {
  const state = { row: { ...initialRow } };
  const builder = (table: string): any => {
    let pendingUpdate: any = null;
    const ctx: any = {};
    ctx.select = () => ctx;
    ctx.eq = () => ctx;
    ctx.or = () => ctx;
    ctx.single = async () => ({ data: state.row, error: null });
    ctx.update = (vals: any) => {
      pendingUpdate = vals;
      return ctx;
    };
    // Trigger update commit when the chain ends without .select()
    ctx.then = (resolve: any) => {
      if (pendingUpdate) {
        Object.assign(state.row, pendingUpdate);
        pendingUpdate = null;
      }
      return Promise.resolve({ data: null, error: null }).then(resolve);
    };
    return ctx;
  };
  return {
    client: { from: builder } as any,
    state,            // tests mutate state.row to simulate rotations
    setRow(next: Partial<XeroConnection>) { Object.assign(state.row, next); },
  };
}
```
