<objective>
Foundation plan for Phase 52. Ships the API + types + helper plumbing that Plans 52-01 (UI) and 52-02 (re-import reconciliation) build on. Three concrete deliverables:

1. **Extend `/api/Xero/employees` route** to fetch `/PayrollCalendars` (1 extra request per import) and join `Employee.PayrollCalendarID` → `CalendarType` so the response carries `pay_frequency` per employee. Also fixes the latent **`EmploymentType` vs `EmploymentBasis`** bug flagged in 52-RESEARCH.md (existing route reads the wrong field name; everyone falls through to "full-time").

2. **New pure helper module** `xero-payroll-mapping.ts` (no I/O, no React) housing every Xero-→-wizard mapping decision in one trivially-testable place. Required by PHASE.md XERO-S4-02.

3. **Extend TeamMember + NewHire types** with the optional Xero-provenance fields the rest of the phase needs (`standardHours`, `_xeroEmployeeId`, `_xeroImportedAt`, `_xeroFingerprint`, `_overriddenFields`). Backfill the existing first-load auto-import in `ForecastWizardV4.tsx` so it stops dropping these fields on the floor. `hourlyRate?` already exists on both types — no change there.

**Why these three pieces in one plan and not split:** the API change is too small to be its own plan (~50 LOC), and Plan 52-01's UI strictly requires the new `pay_frequency` field on the API response. Splitting creates an artificial sequencing constraint with no review benefit. The pure-helper module + type extension are bundled because they have a 1:1 dependency (helper signatures import the new types) and zero behavioural overlap with the UI plan.

**Operator's Option D decision** (auto-detect with override markers + reconcile on re-import) is encoded here at the type layer: `_overriddenFields?: string[]` (kept as an array of field names — JSON-serialisable; a `Set` would not survive localStorage rehydration). Plan 52-01 reads/writes it on import; Plan 52-02 reads it during reconciliation.

**No UI in this plan.** No import modal. No diff banner. Those land in 52-01 and 52-02 respectively.

Output: 1 modified route, 1 new helper module, 1 modified types file, 1 modified wizard file (3 mapper call sites consolidated), 2 new test files. ~3-4 atomic commits. Single PR.
</objective>

## xero-payroll-mapping.ts (NEW)

```typescript
// src/app/finances/forecast/components/wizard-v4/utils/xero-payroll-mapping.ts
// Pure functions — no I/O, no React, no Supabase. Trivially unit-testable.
// All input shapes mirror Xero Payroll AU REST JSON (PascalCase).
// TODO(phase-future): NZ/UK Payroll API uses different schema — currently AU only.

import type {
  PayFrequency,
  EmploymentType,
  TeamMember,
  NewHire,
  XeroFieldFingerprint,
} from '../types';

/**
 * Maps Xero Payroll AU CalendarType → wizard PayFrequency.
 * Verified against node_modules/xero-node/dist/gen/model/payroll-au/calendarType.d.ts.
 * FOURWEEKLY/TWICEMONTHLY/QUARTERLY collapse to 'monthly' (nearest cashflow approximation).
 */
export function mapXeroPayrollCalendarToFrequency(
  calendarType: string | undefined | null,
): PayFrequency | undefined {
  if (!calendarType) return undefined;
  switch (calendarType.toUpperCase()) {
    case 'WEEKLY': return 'weekly';
    case 'FORTNIGHTLY': return 'fortnightly';
    case 'FOURWEEKLY':
    case 'TWICEMONTHLY':
    case 'MONTHLY':
    case 'QUARTERLY': return 'monthly';
    default: return undefined;
  }
}

/**
 * Maps Xero EmploymentBasis (or legacy EmploymentType field) → wizard EmploymentType.
 * Replaces the inline EMPLOYMENT_TYPE_MAP at route.ts:38-44 (which was keyed off
 * the wrong field name — see 52-RESEARCH.md Pitfall 2).
 */
export function normaliseXeroEmployment(
  basis: string | undefined | null,
): EmploymentType {
  if (!basis) return 'full-time';
  switch (basis.toUpperCase()) {
    case 'FULLTIME': return 'full-time';
    case 'PARTTIME': return 'part-time';
    case 'CASUAL': return 'casual';
    case 'CONTRACTOR':
    case 'LABOURHIRE':
    case 'SUPERINCOMESTREAM': return 'contractor';
    case 'NONEMPLOYEE': return 'contractor';  // safe fallback; rarely seen
    default: return 'full-time';
  }
}

/**
 * Classifies a Xero EarningsRateCalculationType.
 * Used by Plan 52-01 to branch the import-modal UI:
 *   - 'hourly' → annual salary cell read-only with edit affordance
 *   - 'salaried' → annual salary editable; hourly/hours shown as derived hints
 */
export function classifyXeroEarningsRateCalculationType(
  calculationType: string | undefined | null,
): 'hourly' | 'salaried' | undefined {
  if (!calculationType) return undefined;
  switch (calculationType.toUpperCase()) {
    case 'USEEARNINGSRATE':
    case 'ENTEREARNINGSRATE': return 'hourly';
    case 'ANNUALSALARY': return 'salaried';
    default: return undefined;
  }
}

/**
 * Extract { hourlyRate, annualSalary, standardHours, calculationType } from a
 * Xero PayTemplate.EarningsLines[]. Matches the first ordinary-earnings line
 * (we only handle OrdinaryEarnings per PHASE.md scope).
 */
export function extractCompensationFromPayTemplate(
  earningsLines: any[] | undefined,
  ordinaryHoursPerWeek: number | undefined,
): {
  hourlyRate?: number;
  annualSalary?: number;
  standardHours?: number;
  calculationType?: 'hourly' | 'salaried';
} {
  if (!earningsLines || earningsLines.length === 0) {
    return { standardHours: ordinaryHoursPerWeek };
  }
  // Pick the first earnings line with a meaningful rate/salary
  for (const line of earningsLines) {
    const calcType = classifyXeroEarningsRateCalculationType(line.CalculationType);
    if (calcType === 'salaried' && line.AnnualSalary != null) {
      return {
        annualSalary: parseFloat(line.AnnualSalary),
        // For salaried staff, OrdinaryHoursPerWeek is the source of truth (not the line)
        standardHours: ordinaryHoursPerWeek,
        calculationType: 'salaried',
      };
    }
    if (calcType === 'hourly' && line.RatePerUnit != null) {
      const standardHours = line.NumberOfUnitsPerWeek != null
        ? parseFloat(line.NumberOfUnitsPerWeek)
        : ordinaryHoursPerWeek;
      return {
        hourlyRate: parseFloat(line.RatePerUnit),
        standardHours,
        calculationType: 'hourly',
      };
    }
  }
  return { standardHours: ordinaryHoursPerWeek };
}

/**
 * The single canonical mapping from a /api/Xero/employees response item to
 * the wizard's TeamMember/NewHire shape. Used by:
 *   - ForecastWizardV4.tsx first-load auto-import (3 call sites)
 *   - Plan 52-01 on-demand "Import from Xero" handler
 *   - Plan 52-02 re-import (via mergeXeroEmployeeIntoMember which calls this)
 */
export interface XeroEmployeeApiShape {
  employee_id: string;
  full_name: string;
  first_name?: string;
  last_name?: string;
  job_title?: string;
  email?: string;
  start_date?: string;
  termination_date?: string;
  annual_salary?: number;
  hourly_rate?: number;
  hours_per_week?: number;
  standard_hours?: number;        // NEW (52-00)
  pay_frequency?: PayFrequency;   // NEW (52-00)
  employment_type?: string;       // already wizard-normalised by route
  calculation_type?: 'hourly' | 'salaried';  // NEW (52-00)
  is_active: boolean;
  from_xero: boolean;
}

export function enrichWizardMemberFromXeroEmployee(
  emp: XeroEmployeeApiShape,
  importedAt: string = new Date().toISOString(),
): Partial<TeamMember> & { name: string; role: string; type: EmploymentType; isFromXero: true } {
  const type = normaliseXeroEmployment(emp.employment_type);
  return {
    name: emp.full_name,
    role: emp.job_title || 'Team member',
    type,
    hoursPerWeek: emp.hours_per_week ?? emp.standard_hours ?? 38,
    hourlyRate: emp.hourly_rate,
    standardHours: emp.standard_hours,
    payFrequency: emp.pay_frequency,
    currentSalary: emp.annual_salary ?? 0,
    isFromXero: true,
    _xeroEmployeeId: emp.employee_id,
    _xeroImportedAt: importedAt,
    _xeroFingerprint: computeXeroFingerprint({
      payFrequency: emp.pay_frequency,
      standardHours: emp.standard_hours,
      hourlyRate: emp.hourly_rate,
      currentSalary: emp.annual_salary,
      hoursPerWeek: emp.hours_per_week,
      type,
      name: emp.full_name,
      role: emp.job_title,
    }),
  };
}

/**
 * Snapshot the current Xero-sourced field values. Used to detect operator edits
 * on re-import (compare current member values vs the fingerprint from last import).
 */
export function computeXeroFingerprint(
  values: Partial<XeroFieldFingerprint>,
): XeroFieldFingerprint {
  // Strip undefined keys for compactness; serialise cleanly to localStorage
  const result: XeroFieldFingerprint = {};
  for (const [k, v] of Object.entries(values)) {
    if (v !== undefined) (result as any)[k] = v;
  }
  return result;
}
```
