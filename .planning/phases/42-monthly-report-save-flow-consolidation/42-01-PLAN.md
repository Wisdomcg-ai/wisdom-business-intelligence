export type SaveStatus =
  | { kind: 'idle' }
  | { kind: 'saving' }
  | { kind: 'saved'; at: Date }
  | { kind: 'retrying'; attempt: 1 | 2 | 3 }
  | { kind: 'failed' }

export interface UseAutoSaveReportArgs {
  report: GeneratedReport | null
  commentary: VarianceCommentary | undefined
  userId: string | null
  isLocked: boolean              // true when monthly_report_snapshots.status === 'final'
  onSaveSuccess?: () => void
  saveSnapshot: (reportData: GeneratedReport, options: SaveSnapshotOptions) => Promise<unknown>
}

export interface UseAutoSaveReportReturn {
  status: SaveStatus
  schedule: () => void           // debounce path (call from onChange)
  flushImmediately: () => void   // blur path (cancel debounce, fire now)
  retryNow: () => void           // user clicks Save Now after retries exhausted
}
