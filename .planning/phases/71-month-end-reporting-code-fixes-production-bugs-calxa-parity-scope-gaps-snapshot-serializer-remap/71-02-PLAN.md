       export function tokenSortKey(name: string): string {
         // 1. lowercase, 2. strip punctuation (replace [^\w\s] with space), 3. split on whitespace,
         // 4. filter empty, 5. sort alphabetically, 6. join with single space.
         return (name || '')
           .toLowerCase()
           .replace(/[^\w\s]/g, ' ')
           .split(/\s+/)
           .filter(Boolean)
           .sort()
           .join(' ')
       }
