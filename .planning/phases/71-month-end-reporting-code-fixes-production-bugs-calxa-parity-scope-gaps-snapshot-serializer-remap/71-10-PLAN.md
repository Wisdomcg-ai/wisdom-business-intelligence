<objective>
D4 — Snapshot serializer fix + data remap migration.

Purpose: Today `report_data.sections` is serialized as a JS array (`ReportSection[]`), so when persisted as JSONB it becomes numeric-keyed (`{"0": {...}, "1": {...}}`). PDF generator and other consumers cannot resolve sections by name (e.g. "wages_detail", "subscription_detail"). Per CONTEXT D-D4:

1. D4-code: serializer writes named keys (e.g. `{ revenue: {...}, cost_of_sales: {...}, operating_expenses: {...} }`).
2. D4-data: one-off migration script remaps existing snapshot rows from numeric → named keys (idempotent, dry-run first).

Output: Patched serializer + new regression test + new backfill script with explicit human checkpoint before --apply.
</objective>

       ```typescript
       import type { ReportSection } from '../types'

       export const CATEGORY_KEY_MAP: Record<string, string> = {
         'Revenue': 'revenue',
         'Cost of Sales': 'cost_of_sales',
         'Operating Expenses': 'operating_expenses',
         'Other Income': 'other_income',
         'Other Expenses': 'other_expenses',
       }

       const KEY_TO_CATEGORY: Record<string, string> = Object.fromEntries(
         Object.entries(CATEGORY_KEY_MAP).map(([k, v]) => [v, k]),
       )

       export type NamedSectionMap = Record<string, ReportSection>

       export function categoryToKey(category: string): string {
         return CATEGORY_KEY_MAP[category] ?? category.toLowerCase().replace(/\s+/g, '_')
       }

       export function serializeReportSections(
         input: ReportSection[] | NamedSectionMap,
       ): NamedSectionMap {
         if (!Array.isArray(input)) return input  // already named — idempotent
         const out: NamedSectionMap = {}
         for (const sec of input) {
           out[categoryToKey(sec.category)] = sec
         }
         return out
       }

       export function deserializeReportSections(
         input: NamedSectionMap | ReportSection[],
       ): ReportSection[] {
         if (Array.isArray(input)) return input  // already array
         // Preserve CATEGORY_ORDER from useMonthlyReport — re-import or inline:
         const order = ['Revenue', 'Cost of Sales', 'Operating Expenses', 'Other Income', 'Other Expenses']
         const named = input as NamedSectionMap
         const arr: ReportSection[] = []
         for (const category of order) {
           const key = categoryToKey(category)
           if (named[key]) arr.push(named[key])
         }
         // Append any keys not in the standard order (passthrough)
         for (const [key, sec] of Object.entries(named)) {
           if (!order.some(c => categoryToKey(c) === key)) arr.push(sec)
         }
         return arr
       }
       ```

    ```javascript
    #!/usr/bin/env node
    // 71-D4 — Snapshot sections remap (numeric → named keys)
    // Usage:
    //   node scripts/71-D4-snapshot-sections-remap.mjs            # dry-run
    //   node scripts/71-D4-snapshot-sections-remap.mjs --apply    # commit writes

    import { createClient } from '@supabase/supabase-js'

    const APPLY = process.argv.includes('--apply')
    const MODE = APPLY ? 'APPLY' : 'DRY-RUN'

    const supabase = createClient(
      process.env.NEXT_PUBLIC_SUPABASE_URL,
      process.env.SUPABASE_SERVICE_ROLE_KEY,
    )

    // Duplicate of CATEGORY_KEY_MAP from src/.../snapshot-serializer.ts (kept in sync manually — single small map).
    const CATEGORY_KEY_MAP = {
      'Revenue': 'revenue',
      'Cost of Sales': 'cost_of_sales',
      'Operating Expenses': 'operating_expenses',
      'Other Income': 'other_income',
      'Other Expenses': 'other_expenses',
    }
    const categoryToKey = (c) => CATEGORY_KEY_MAP[c] ?? c.toLowerCase().replace(/\s+/g, '_')

    function isNumericKeyedSections(sections) {
      if (!sections || typeof sections !== 'object' || Array.isArray(sections)) return Array.isArray(sections)
      const keys = Object.keys(sections)
      return keys.length > 0 && keys.every(k => /^\d+$/.test(k))
    }

    function remapSections(sections) {
      // input shape: { "0": { category: 'Revenue', ... }, "1": { ... } }  or array
      const arr = Array.isArray(sections) ? sections : Object.values(sections)
      const out = {}
      for (const sec of arr) {
        if (sec?.category) out[categoryToKey(sec.category)] = sec
      }
      return out
    }

    async function main() {
      console.log(`[71-D4] Mode: ${MODE}`)
      const { data: rows, error } = await supabase
        .from('monthly_report_snapshots')
        .select('id, business_id, report_month, report_data')
      if (error) { console.error(error); process.exit(1) }
      console.log(`[71-D4] Loaded ${rows.length} snapshot rows`)

      let needRemap = 0, alreadyNamed = 0, skipped = 0, applied = 0, failed = 0

      for (const row of rows) {
        const sections = row.report_data?.sections
        if (!sections) { skipped++; continue }
        const needs = isNumericKeyedSections(sections) || Array.isArray(sections)
        if (!needs) { alreadyNamed++; continue }
        needRemap++

        const newSections = remapSections(sections)
        const newReportData = { ...row.report_data, sections: newSections }

        console.log(`  - id=${row.id} biz=${row.business_id} month=${row.report_month}: ${Object.keys(sections).length} keys → ${Object.keys(newSections).join(', ')}`)

        if (APPLY) {
          const { error: upErr } = await supabase
            .from('monthly_report_snapshots')
            .update({ report_data: newReportData })
            .eq('id', row.id)
          if (upErr) { console.error(`    FAILED: ${upErr.message}`); failed++ }
          else applied++
        }
      }

      console.log(`\n[71-D4] Summary: ${rows.length} total | ${needRemap} need remap | ${alreadyNamed} already named | ${skipped} skipped (no sections)`)
      if (APPLY) console.log(`[71-D4] Applied: ${applied} | Failed: ${failed}`)
      else console.log(`[71-D4] DRY-RUN complete. Re-run with --apply to commit.`)
    }

    main().catch(e => { console.error(e); process.exit(1) })
    ```

    Idempotency: re-running after --apply should report `0 need remap | N already named`.
