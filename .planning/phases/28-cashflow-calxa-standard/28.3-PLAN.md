# Phase 28.3 — Schedule + Distribution Model

## Objective

Replace the current string-based frequency enums (`"quarterly"`, `"monthly"`)
with Calxa's precise BasePeriods[12] arrays. Add per-account Type 1-5 profiles
so individual accounts can override the global DSO/DPO with precise timing.

```ts
// Calxa's days → distribution formula
export function daysToDistribution(days: number): number[] {
  const dist = new Array(12).fill(0)
  const bucket = Math.floor(days / 30)
  const fraction = (days % 30) / 30
  dist[bucket] = Math.round((1 - fraction) * 100)
  dist[bucket + 1] = Math.round(fraction * 100)
  // Normalise to exactly 100
  const sum = dist.reduce((a, b) => a + b, 0)
  if (sum !== 100) dist[bucket] += (100 - sum)
  return dist
}
```

### Engine integration

Replace in engine.ts:
```ts
// OLD
const dsoSplit = getTimingSplit(assumptions.dso_days)

// NEW
function getAccountTimingSplit(line: PLLine, settings, profiles): number[] {
  const profile = profiles.find(p => p.xero_account_id === line.xero_account_id)
  if (profile) {
    switch (profile.cashflow_type) {
      case 1: return [100, 0, 0, ...]  // Immediate
      case 2: return profile.distribution  // Precomputed
      case 3: return daysToDistribution(profile.days)  // Creditor days
      case 4: return daysToDistribution(profile.days)  // Debtor days
      case 5: return scheduleToSplit(profile.schedule_base_periods, currentMonth)
    }
  }
  // Fallback: global DSO/DPO
  return line.category === 'Revenue' ? daysToDistribution(settings.dso_days) : daysToDistribution(settings.dpo_days)
}
```

Replace frequency string lookups:
```ts
// OLD
if (gst_reporting_frequency === 'quarterly' && isBASPaymentMonth(monthNum)) { pay GST }

// NEW
const gstSchedule = await resolveSchedule(settings.gst_schedule, businessId)
if (getPaymentMonth(monthNum, gstSchedule) === monthNum) { pay GST }
```
