```typescript
// Pre-expiry early-warning sensor (Phase 69-04).
// Fires when expires_at is within 24h AND cron did not produce a fresh access
// token this run. Distinct from cron_refresh_xero_tokens_failed (which fires
// only on transient failures); this captures the case where cron "succeeded"
// at the call level but the underlying token is about to die — e.g. token-
// manager short-circuited as still_valid because the row reports a recent
// refresh, but the row's expires_at is closer to now than the next cron tick.
//
// Honours the 53-05 "one event per failure" invariant: this is a DIFFERENT
// signal (pre-expiry observation), not a duplicate of the failure event.
const lastResult = results[results.length - 1]
if (lastResult && lastResult.status !== 'refreshed' && lastResult.status !== 'deactivated') {
  const expiresAtMs = new Date(row.expires_at).getTime()
  const msUntilExpiry = expiresAtMs - Date.now()
  if (msUntilExpiry > 0 && msUntilExpiry < PRE_EXPIRY_WARNING_MS) {
    const hoursUntilExpiry = Math.floor(msUntilExpiry / (60 * 60 * 1000))
    try {
      Sentry.captureMessage('Xero token within 24h of expiry — cron did not refresh', {
        level: 'warning',
        tags: {
          invariant: 'xero_token_pre_expiry',
          connection_id: row.id,
          business_id: row.business_id,
          tenant_id: row.tenant_id,
          hours_until_expiry: String(hoursUntilExpiry),
          last_status: lastResult.status,
        },
        extra: {
          expires_at: row.expires_at,
          tenant_name: row.tenant_name,
        },
      } as any)
    } catch {
      // Sentry outage must never abort the cron run.
    }
  }
}
```
