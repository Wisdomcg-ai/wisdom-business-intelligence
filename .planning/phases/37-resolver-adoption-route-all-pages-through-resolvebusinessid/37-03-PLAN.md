```typescript
export async function resolveBusinessId(
  supabase: SupabaseClient,
  params: {
    userId: string | null | undefined
    role: 'client' | 'coach' | 'admin' | null | undefined
    activeBusinessId: string | null | undefined
  }

```

existing empty-state branch already handles.

That is a SEPARATE concern after id resolution — keep that follow-up query.
