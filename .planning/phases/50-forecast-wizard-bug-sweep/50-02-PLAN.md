    // - When item.lease_type is set: use new accounting math (4 branches)
    // - When item.lease_type is undefined: fall through to legacy paymentMethod logic
    export function getPlannedSpendPLImpact(item: PlannedSpend, yearNum: 1 | 2 | 3): number {
      if (item.lease_type) {
        return getPlannedSpendPLImpactWithTaxonomy(item, yearNum);
      }
      return getPlannedSpendPLImpactLegacy(item, yearNum);
    }

    function getPlannedSpendPLImpactWithTaxonomy(item: PlannedSpend, yearNum: 1 | 2 | 3): number {
      const monthsRemaining = yearNum === 1 ? Math.max(0, 13 - item.month) : 12;
      const depreciableBase = item.amount - (item.residual_value || 0);
      const usefulLifeMonths = item.useful_life_months || 0;
      const termMonths = item.term_months || 0;

      switch (item.lease_type) {
        case 'outright_purchase': {
          if (usefulLifeMonths <= 0) return 0;
          const monthlyDep = depreciableBase / usefulLifeMonths;
          return Math.round(monthlyDep * monthsRemaining);
        }
        case 'operating_lease': {
          const monthlyPayment = item.leaseMonthlyPayment
            ?? (termMonths > 0 ? item.amount / termMonths : 0);
          return Math.round(monthlyPayment * 12);
        }
        case 'finance_lease':
        case 'loan_financing': {

          if (usefulLifeMonths > 0) {

            depPortion = monthlyDep * monthsRemaining;
