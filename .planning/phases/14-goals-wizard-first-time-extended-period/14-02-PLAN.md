```typescript
/**
 * Get the display label for a year column.
 * When extended period is active, Year 1 shows a combined label like
 * "Apr-Jun 2026 + FY2027 (14 months)" instead of just "FY2027".
 */
export function getYearLabel(
  yearType: YearType,
  yearNum: number,
  extendedPeriodInfo?: { isExtendedPeriod: boolean; year1Months: number; currentYearRemainingMonths: number }
): string {
  const currentYear = new Date().getFullYear()
  const currentMonth = new Date().getMonth() + 1

  if (yearType === 'FY') {
    const fiscalYear = currentMonth >= 7 ? currentYear + 1 : currentYear
    const targetFY = fiscalYear + yearNum - 1

    // Extended period: Year 1 spans current FY remainder + next full FY
    if (yearNum === 1 && extendedPeriodInfo?.isExtendedPeriod) {
      return `FY${targetFY - 1} remainder + FY${targetFY} (${extendedPeriodInfo.year1Months}mo)`
    }

    return `FY${targetFY}`
  } else {
    const targetCY = currentMonth >= 11 ? currentYear + yearNum : currentYear + yearNum - 1

    if (yearNum === 1 && extendedPeriodInfo?.isExtendedPeriod) {
      return `CY${targetCY - 1} remainder + CY${targetCY} (${extendedPeriodInfo.year1Months}mo)`
    }

    return `CY${targetCY}`
  }
}
```
