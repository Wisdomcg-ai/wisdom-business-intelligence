## Modal editor sketch (Task 5)

Inline conditional render matching `showAddRevenue` pattern:

```tsx
{showSeasonalityFor && (
  <SeasonalityEditorModal
    line={state.revenueLines.find(l => l.id === showSeasonalityFor)
        ?? state.cogsLines.find(l => l.id === showSeasonalityFor)}
    businessSeasonality={businessSeasonality}
    fiscalYearStart={fiscalYearStart}
    onSave={(pattern) => {
      const isRevenue = state.revenueLines.some(l => l.id === showSeasonalityFor);
      if (isRevenue) actions.updateRevenueLine(showSeasonalityFor, { seasonalityPattern: pattern });
      else actions.updateCOGSLine(showSeasonalityFor, { seasonalityPattern: pattern });
      setShowSeasonalityFor(null);
    }}
    onReset={() => {
      const isRevenue = state.revenueLines.some(l => l.id === showSeasonalityFor);
      if (isRevenue) actions.updateRevenueLine(showSeasonalityFor, { seasonalityPattern: undefined });
      else actions.updateCOGSLine(showSeasonalityFor, { seasonalityPattern: undefined });
      setShowSeasonalityFor(null);
    }}
    onCancel={() => setShowSeasonalityFor(null)}
  />
)}
```
