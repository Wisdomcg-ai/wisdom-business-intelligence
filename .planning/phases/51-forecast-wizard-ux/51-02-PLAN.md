```tsx
interface RevenueLineMixInputsProps {
  lineId: string;
  lineName: string;
  lineTotal: number;
  linePct: number;
  onCommitDollar: (value: number) => void;
  onCommitPct: (value: number) => void;
  size?: 'sm' | 'xs';

  // Phase 51 plan 02 (UX-S3-02) — Growth % editor for Y2/Y3

  // When activeYear === 2 or 3, growth editor is rendered with growthPct as the
  // committed value and onCommitGrowth as the commit handler.
  activeYear?: 1 | 2 | 3;       // default 1 (back-compat with 51-01 call sites)

```

The `growthPct` displayed value is computed by the parent (Step3RevenueCOGS) as:

```typescript
const growthPct = priorYearTotal > 0
```

The commit handler is:

```typescript
const commitGrowthValue = (lineId: string, growthValue: number, year: 2 | 3) => {
  const clamped = Math.max(-100, Math.min(1000, growthValue));
  handleGrowthChange(lineId, clamped, year);
};
```
