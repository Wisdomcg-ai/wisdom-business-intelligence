**Function `loadRows(forecastId: string): Promise<Row[]>`:**
```typescript
async function loadRows(forecastId: string): Promise<Row[]> {
  const { data, error } = await supabase
    .from('forecast_pl_lines')
    .select('id, account_code, account_name, category, is_manual, forecast_months, actual_months, computed_at')
    .eq('forecast_id', forecastId)
    .order('id', { ascending: true })
  if (error) throw new Error(`loadRows failed: ${error.message}`)
  return (data ?? []) as Row[]
}
```

**Snapshot mode:**
```typescript
async function snapshot() {
  const rows = await loadRows(forecastId!)
  const snap: Snapshot = {
    forecast_id: forecastId!,
    snapshot_taken_at: new Date().toISOString(),
    rows,
  }
  const filename = args['snapshot-file'] || join(SNAPSHOT_DIR, `canary-snapshot-${forecastId}-${snap.snapshot_taken_at.replace(/[:.]/g, '-')}.json`)
  writeFileSync(filename, JSON.stringify(snap, null, 2))
  console.log(`[CANARY] Snapshot saved: ${filename}`)
  console.log(`[CANARY] Rows captured: ${rows.length}`)
  console.log(`[CANARY] is_manual=true: ${rows.filter((r) => r.is_manual).length}`)
  console.log(`[CANARY] is_manual=false: ${rows.filter((r) => !r.is_manual).length}`)
  console.log(`[CANARY] account_codes captured: ${[...new Set(rows.map((r) => r.account_code ?? 'NULL'))].join(', ')}`)
}
```

**Diff mode:**
```typescript
async function diff() {
  const snapFile = args['snapshot-file']
  if (!snapFile || !existsSync(snapFile)) {
    console.error(`[CANARY] --snapshot-file path is required and must exist for --mode=diff`)
    process.exit(1)
  }
  const before = JSON.parse(readFileSync(snapFile, 'utf8')) as Snapshot
  const after = await loadRows(forecastId!)

  console.log(`[CANARY] Diff for forecast=${forecastId}`)
  console.log(`[CANARY] Before snapshot: ${before.snapshot_taken_at} (${before.rows.length} rows)`)
  console.log(`[CANARY] After snapshot:  ${new Date().toISOString()} (${after.rows.length} rows)`)

  // Map by id for stable comparison
  const beforeById = new Map(before.rows.map((r) => [r.id, r]))
  const afterById = new Map(after.map((r) => [r.id, r]))

  // Test 1: row count by is_manual flag
  const beforeManual = before.rows.filter((r) => r.is_manual).length
  const afterManual = after.filter((r) => r.is_manual).length
  const beforeDerived = before.rows.filter((r) => !r.is_manual).length
  const afterDerived = after.filter((r) => !r.is_manual).length

  // Test 2: rows lost — derived rows present before but absent after by id
  const lostDerivedIds: string[] = []
  for (const [id, row] of beforeById) {
    if (!row.is_manual && !afterById.has(id)) {
      // Allow loss only if account_code is present in afterById under a different id
      // (re-insert path — UPSERT replaces, doesn't carry id, but the partial unique
      // index conflict target is account_code, so on conflict the row keeps its id;
      // a missing id therefore IS a row loss).
      lostDerivedIds.push(id)
    }
  }

  // Test 3: blanked forecast_months — compare same id before/after
  const blanked: string[] = []
  for (const [id, beforeRow] of beforeById) {
    if (beforeRow.is_manual) continue
    const afterRow = afterById.get(id)
    if (!afterRow) continue
    const beforeKeys = Object.keys(beforeRow.forecast_months ?? {}).length
    const afterKeys = Object.keys(afterRow.forecast_months ?? {}).length
    if (beforeKeys > 0 && afterKeys === 0) blanked.push(id)
  }

  // Test 4: manual rows untouched
  const manualMutated: string[] = []
  for (const [id, beforeRow] of beforeById) {
    if (!beforeRow.is_manual) continue
    const afterRow = afterById.get(id)
    if (!afterRow) {
      manualMutated.push(`${id} (LOST)`)
      continue
    }
    if (JSON.stringify(beforeRow.forecast_months) !== JSON.stringify(afterRow.forecast_months)) {
      manualMutated.push(`${id} (forecast_months changed)`)
    }
    if (beforeRow.computed_at !== afterRow.computed_at) {
      manualMutated.push(`${id} (computed_at bumped)`)
    }
  }

  // Test 5: computed_at moved forward on derived rows
  const stale: string[] = []
  for (const [id, beforeRow] of beforeById) {
    if (beforeRow.is_manual) continue
    const afterRow = afterById.get(id)
    if (!afterRow) continue
    if ((afterRow.computed_at ?? '') <= (beforeRow.computed_at ?? '')) {
      stale.push(id)
    }
  }

  console.log('')
  console.log('=== CANARY DIFF RESULT ===')
  console.log(`[T1] Manual rows: before=${beforeManual} after=${afterManual} ${beforeManual === afterManual ? 'PASS' : 'FAIL'}`)
  console.log(`[T1] Derived rows: before=${beforeDerived} after=${afterDerived} (allowed to grow; never shrink unexpectedly)`)
  console.log(`[T2] Derived rows lost (by id): ${lostDerivedIds.length} ${lostDerivedIds.length === 0 ? 'PASS' : 'FAIL'}`)
  if (lostDerivedIds.length) console.log(`     IDs: ${lostDerivedIds.join(', ')}`)
  console.log(`[T3] Derived rows with blanked forecast_months: ${blanked.length} ${blanked.length === 0 ? 'PASS' : 'FAIL'}`)
  if (blanked.length) console.log(`     IDs: ${blanked.join(', ')}`)
  console.log(`[T4] Manual rows mutated: ${manualMutated.length} ${manualMutated.length === 0 ? 'PASS' : 'FAIL'}`)
  if (manualMutated.length) console.log(`     ${manualMutated.join('; ')}`)
  console.log(`[T5] Derived rows with stale computed_at: ${stale.length} ${stale.length === 0 ? 'PASS' : 'FAIL'}`)
  if (stale.length) console.log(`     IDs: ${stale.join(', ')}`)

  const overall = lostDerivedIds.length === 0 && blanked.length === 0 && manualMutated.length === 0 && stale.length === 0 && beforeManual === afterManual
  console.log('')
  console.log(`=== OVERALL: ${overall ? 'PASS — proceed to push main per D-44.1-20 step (1)' : 'FAIL — DO NOT PUSH; investigate'} ===`)
  process.exit(overall ? 0 : 1)
}
```
